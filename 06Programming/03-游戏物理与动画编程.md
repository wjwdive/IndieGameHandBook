## 1、碰撞检测基础

### 1.1 碰撞检测的重要性

```python
def circle_vs_circle(c1, c2):
    # 计算圆心距离
    dx = c2.center_x - c1.center_x
    dy = c2.center_y - c1.center_y
    distance_squared = dx * dx + dy * dy

    # 比较距离与半径之和
    radius_sum = c1.radius + c2.radius
    return distance_squared <= radius_sum * radius_sum
```

```python
def get_all_axes(poly1, poly2):
    axes = []

    # 获取多边形1的边法向量
    for i in range(len(poly1.vertices)):
        p1 = poly1.vertices[i]
        p2 = poly1.vertices[(i + 1) % len(poly1.vertices)]
        edge = p2 - p1
        normal = Vector2(-edge.y, edge.x).normalize()
        axes.append(normal)

    # 获取多边形2的边法向量
    for i in range(len(poly2.vertices)):
        p1 = poly2.vertices[i]
        p2 = poly2.vertices[(i + 1) % len(poly2.vertices)]
        edge = p2 - p1
        normal = Vector2(-edge.y, edge.x).normalize()
        axes.append(normal)

    return axes
```

```python
class Grid:
    def __init__(self, cell_size, width, height):
        self.cell_size = cell_size
        self.cols = math.ceil(width / cell_size)
        self.rows = math.ceil(height / cell_size)
        self.cells = [[[] for _ in range(self.cols)] for _ in range(self.rows)]

    def add_object(self, obj):
        # 计算对象所在的网格单元
        start_col = int(obj.aabb.min_x / self.cell_size)
        end_col = int(obj.aabb.max_x / self.cell_size)
        start_row = int(obj.aabb.min_y / self.cell_size)
        end_row = int(obj.aabb.max_y / self.cell_size)

        # 添加到所有覆盖的单元
        for row in range(start_row, end_row + 1):
            for col in range(start_col, end_col + 1):
                if 0 <= row < self.rows and 0 <= col < self.cols:
                    self.cells[row][col].append(obj)

    def get_potential_collisions(self, obj):
        potential = set()

        # 获取对象所在网格单元
        start_col = int(obj.aabb.min_x / self.cell_size)
        end_col = int(obj.aabb.max_x / self.cell_size)
        start_row = int(obj.aabb.min_y / self.cell_size)
        end_row = int(obj.aabb.max_y / self.cell_size)

        # 收集所有相邻单元中的对象
        for row in range(start_row, end_row + 1):
            for col in range(start_col, end_col + 1):
                if 0 <= row < self.rows and 0 <= col < self.cols:
                    for other in self.cells[row][col]:
                        if other != obj:
                            potential.add(other)

        return list(potential)
```

## 2、刚体物理模拟

### 2.1 基本物理概念

#### 牛顿运动定律在游戏中的应用
1. **惯性**：物体保持运动状态
2. **力与加速度**：F = ma
3. **作用与反作用**：碰撞时的力相互作用

```python
class RigidBody:
    def __init__(self, mass=1.0):
        self.mass = mass
        self.position = Vector2(0, 0)
        self.velocity = Vector2(0, 0)
        self.acceleration = Vector2(0, 0)
        self.force = Vector2(0, 0)

        # 旋转相关
        self.rotation = 0
        self.angular_velocity = 0
        self.angular_acceleration = 0
        self.torque = 0
        self.inertia = mass * 1.0  # 简化惯性矩

    def apply_force(self, force, point=None):
        # 应用力到质心
        self.force += force

        # 如果指定了作用点，计算扭矩
        if point is not None:
            r = point - self.position
            self.torque += r.cross(force)

    def update(self, delta_time):
        # 线性运动
        self.acceleration = self.force / self.mass
        self.velocity += self.acceleration * delta_time
        self.position += self.velocity * delta_time

        # 角运动
        self.angular_acceleration = self.torque / self.inertia
        self.angular_velocity += self.angular_acceleration * delta_time
        self.rotation += self.angular_velocity * delta_time

        # 重置力
        self.force = Vector2(0, 0)
        self.torque = 0
```

### 2.2 碰撞响应

#### 冲量法（Impulse Method）
```python
def resolve_collision(body_a, body_b, normal, depth):
    # 计算相对速度
    relative_velocity = body_b.velocity - body_a.velocity

    # 沿法线方向的相对速度
    velocity_along_normal = relative_velocity.dot(normal)

    # 如果物体正在分离，不处理
    if velocity_along_normal > 0:
        return

    # 计算恢复系数（弹性）
    e = min(body_a.restitution, body_b.restitution)

    # 计算冲量大小
    j = -(1 + e) * velocity_along_normal
    j /= body_a.inv_mass + body_b.inv_mass

    # 应用冲量
    impulse = normal * j
    body_a.velocity -= impulse * body_a.inv_mass
    body_b.velocity += impulse * body_b.inv_mass

    # 位置修正（防止穿透）
    percent = 0.2  # 通常20%-80%
    slop = 0.01  # 允许的穿透量
    correction = max(depth - slop, 0.0) / (body_a.inv_mass + body_b.inv_mass) * percent
    correction_vector = normal * correction

    body_a.position -= correction_vector * body_a.inv_mass
    body_b.position += correction_vector * body_b.inv_mass
```

#### 摩擦力
```python
def apply_friction(body_a, body_b, normal, contact_point):
    # 计算相对速度在接触点
    ra = contact_point - body_a.position
    rb = contact_point - body_b.position
    relative_velocity = (body_b.velocity + rb.cross(body_b.angular_velocity) -
                        body_a.velocity - ra.cross(body_a.angular_velocity))

    # 切线方向
    tangent = relative_velocity - normal * relative_velocity.dot(normal)
    if tangent.length() > 0:
        tangent = tangent.normalize()

    # 计算摩擦力冲量
    jt = -relative_velocity.dot(tangent)
    jt /= body_a.inv_mass + body_b.inv_mass

    # 库仑摩擦定律
    mu = math.sqrt(body_a.friction * body_b.friction)
    friction_impulse = tangent * jt
    max_friction = mu * jt

    if friction_impulse.length() > max_friction:
        friction_impulse = friction_impulse.normalize() * max_friction

    # 应用摩擦力
    body_a.velocity -= friction_impulse * body_a.inv_mass
    body_b.velocity += friction_impulse * body_b.inv_mass
```

### 2.3 约束和关节

#### 距离约束
```python
class DistanceConstraint:
    def __init__(self, body_a, body_b, distance):
        self.body_a = body_a
        self.body_b = body_b
        self.target_distance = distance

    def solve(self):
        # 计算当前距离
        delta = self.body_b.position - self.body_a.position
        current_distance = delta.length()

        # 计算修正方向
        if current_distance > 0:
            correction = delta * (current_distance - self.target_distance) / current_distance

            # 应用位置修正
            total_inv_mass = self.body_a.inv_mass + self.body_b.inv_mass
            self.body_a.position += correction * (self.body_a.inv_mass / total_inv_mass)
            self.body_b.position -= correction * (self.body_b.inv_mass / total_inv_mass)
```

#### 旋转关节
```python
class RevoluteJoint:
    def __init__(self, body_a, body_b, anchor):
        self.body_a = body_a
        self.body_b = body_b
        self.anchor = anchor  # 世界坐标中的锚点

    def solve(self):
        # 计算局部锚点
        local_a = self.anchor - self.body_a.position
        local_b = self.anchor - self.body_b.position

        # 位置误差
        error = (self.body_b.position + local_b) - (self.body_a.position + local_a)

        # 应用位置修正
        total_inv_mass = self.body_a.inv_mass + self.body_b.inv_mass
        correction = error * 0.5

        self.body_a.position += correction * self.body_a.inv_mass / total_inv_mass
        self.body_b.position -= correction * self.body_b.inv_mass / total_inv_mass
```

## 3、动画系统

### 3.1 精灵动画（Sprite Animation）

#### 帧动画
```python
class SpriteAnimation:
    def __init__(self, frames, frame_duration, loop=True):
        self.frames = frames  # 纹理列表
        self.frame_duration = frame_duration  # 每帧持续时间（秒）
        self.loop = loop

        self.current_time = 0
        self.current_frame = 0
        self.is_playing = True

    def update(self, delta_time):
        if not self.is_playing:
            return

        self.current_time += delta_time

        # 计算当前帧
        total_frames = len(self.frames)
        frame_index = int(self.current_time / self.frame_duration)

        if self.loop:
            self.current_frame = frame_index % total_frames
        else:
            self.current_frame = min(frame_index, total_frames - 1)
            if self.current_frame == total_frames - 1:
                self.is_playing = False

    def get_current_frame(self):
        return self.frames[self.current_frame]

    def reset(self):
        self.current_time = 0
        self.current_frame = 0
        self.is_playing = True
```

#### 动画状态机
```python
class AnimationStateMachine:
    def __init__(self):
        self.states = {}
        self.current_state = None
        self.transitions = {}

    def add_state(self, name, animation):
        self.states[name] = animation

    def add_transition(self, from_state, to_state, condition):
        if from_state not in self.transitions:
            self.transitions[from_state] = []
        self.transitions[from_state].append((to_state, condition))

    def update(self, delta_time, context):
        # 更新当前状态
        if self.current_state:
            self.states[self.current_state].update(delta_time)

        # 检查状态转换
        if self.current_state in self.transitions:
            for to_state, condition in self.transitions[self.current_state]:
                if condition(context):
                    self.change_state(to_state)
                    break

    def change_state(self, new_state):
        if self.current_state:
            self.states[self.current_state].reset()

        self.current_state = new_state
        if new_state in self.states:
            self.states[new_state].reset()
```

### 3.2 骨骼动画（Skeletal Animation）

#### 骨骼和关节
```python
class Bone:
    def __init__(self, name, length, rotation=0, translation=(0, 0)):
        self.name = name
        self.length = length
        self.local_rotation = rotation
        self.local_translation = Vector2(*translation)
        self.parent = None
        self.children = []

        # 变换矩阵
        self.local_matrix = Matrix3.identity()
        self.world_matrix = Matrix3.identity()

    def update_transform(self, parent_world_matrix=None):
        # 计算局部变换矩阵
        self.local_matrix = (Matrix3.translation(self.local_translation) *
                            Matrix3.rotation(self.local_rotation))

        # 计算世界变换矩阵
        if parent_world_matrix:
            self.world_matrix = parent_world_matrix * self.local_matrix
        else:
            self.world_matrix = self.local_matrix

        # 更新子骨骼
        for child in self.children:
            child.update_transform(self.world_matrix)
```

#### 动画关键帧
```python
class Keyframe:
    def __init__(self, time, rotation, translation):
        self.time = time
        self.rotation = rotation
        self.translation = translation

class BoneAnimation:
    def __init__(self, bone_name):
        self.bone_name = bone_name
        self.keyframes = []  # 按时间排序的Keyframe列表

    def get_transform_at_time(self, time):
        if not self.keyframes:
            return Matrix3.identity()

        # 找到当前时间前后的关键帧
        prev_frame = None
        next_frame = None

        for i, frame in enumerate(self.keyframes):
            if frame.time <= time:
                prev_frame = frame
            if frame.time >= time:
                next_frame = frame
                break

        # 如果只有单个关键帧或时间超出范围
        if prev_frame is None:
            return Matrix3.identity()
        if next_frame is None:
            return Matrix3.translation(prev_frame.translation) * Matrix3.rotation(prev_frame.rotation)

        # 线性插值
        t = (time - prev_frame.time) / (next_frame.time - prev_frame.time)

        # 旋转插值（使用四元数或角度插值）
        rotation = lerp_angle(prev_frame.rotation, next_frame.rotation, t)

        # 位置插值
        translation = lerp(prev_frame.translation, next_frame.translation, t)

        return Matrix3.translation(translation) * Matrix3.rotation(rotation)
```

### 3.3 程序动画（Procedural Animation）

#### 逆运动学（Inverse Kinematics）
```python
class CCDSolver:  # 循环坐标下降法
    def __init__(self, tolerance=0.01, max_iterations=10):
        self.tolerance = tolerance
        self.max_iterations = max_iterations

    def solve(self, bones, target):
        end_effector = bones[-1]

        for iteration in range(self.max_iterations):
            # 从末端效应器向根节点遍历
            for i in range(len(bones) - 1, -1, -1):
                bone = bones[i]

                # 计算当前末端位置
                current_end = end_effector.world_position()

                # 计算当前关节到末端和目标的向量
                to_end = current_end - bone.world_position()
                to_target = target - bone.world_position()

                # 计算需要旋转的角度
                angle = math.atan2(
                    to_end.x * to_target.y - to_end.y * to_target.x,
                    to_end.x * to_target.x + to_end.y * to_target.y
                )

                # 应用旋转
                bone.local_rotation += angle
                bone.update_transform()

                # 检查是否达到目标
                if (current_end - target).length() < self.tolerance:
                    return True

        return False
```

#### 弹簧物理动画
```python
class Spring:
    def __init__(self, stiffness=100.0, damping=10.0, mass=1.0):
        self.stiffness = stiffness  # 刚度系数
        self.damping = damping      # 阻尼系数
        self.mass = mass            # 质量

        self.position = Vector2(0, 0)
        self.velocity = Vector2(0, 0)
        self.target = Vector2(0, 0)

    def update(self, delta_time):
        # 计算弹簧力 (Hooke's Law: F = -k * x)
        displacement = self.position - self.target
        spring_force = -self.stiffness * displacement

        # 计算阻尼力 (F = -c * v)
        damping_force = -self.damping * self.velocity

        # 总力
        total_force = spring_force + damping_force

        # 计算加速度 (F = ma)
        acceleration = total_force / self.mass

        # 更新速度
        self.velocity += acceleration * delta_time

        # 更新位置
        self.position += self.velocity * delta_time

        return self.position
```

## 4、物理与动画集成

### 4.1 布娃娃系统（Ragdoll）

```python
class Ragdoll:
    def __init__(self):
        self.bodies = []
        self.constraints = []
        self.bone_mapping = {}  # 骨骼到刚体的映射

    def create_from_skeleton(self, skeleton):
        # 为每个骨骼创建刚体
        for bone in skeleton.bones:
            # 根据骨骼长度创建刚体
            length = bone.length
            body = RigidBody(mass=1.0)
            body.position = bone.world_position()

            self.bodies.append(body)
            self.bone_mapping[bone.name] = body

        # 添加约束（模拟关节）
        for bone in skeleton.bones:
            if bone.parent:
                parent_body = self.bone_mapping[bone.parent.name]
                child_body = self.bone_mapping[bone.name]

                # 创建距离约束
                constraint = DistanceConstraint(parent_body, child_body, bone.length)
                self.constraints.append(constraint)

    def update(self, delta_time):
        # 更新物理
        for body in self.bodies:
            body.apply_force(Vector2(0, 9.8 * body.mass))  # 重力
            body.update(delta_time)

        # 解决约束
        for _ in range(3):  # 多次迭代提高精度
            for constraint in self.constraints:
                constraint.solve()

    def apply_to_skeleton(self, skeleton):
        # 将物理状态应用到骨骼
        for bone_name, body in self.bone_mapping.items():
            bone = skeleton.get_bone(bone_name)
            if bone:
                bone.world_position = body.position
                bone.world_rotation = body.rotation
```

```python
class AnimationDrivenPhysics:
    def __init__(self, character):
        self.character = character
        self.physics_body = RigidBody(mass=70.0)  # 假设角色70kg
        self.animation_body = None  # 动画目标位置

        animation_force = (spring_position - self.physics_body.position) * 1000.0

        # 3. 应用力到物理体
        self.physics_body.apply_force(animation_force)

        # 4. 应用重力和其他力
        self.physics_body.apply_force(Vector2(0, 9.8 * self.physics_body.mass))

        # 5. 更新物理
        self.physics_body.update(delta_time)

        # 6. 最终角色位置是物理和动画的混合
        blend_factor = 0.7  # 70%物理，30%动画
        final_position = (self.physics_body.position * blend_factor +
                         spring_position * (1 - blend_factor))

        self.character.position = final_position
```

### 4.3 脚步IK（Foot IK）

```python
class FootIKSystem:
    def __init__(self, left_foot_bone, right_foot_bone):
        self.left_foot = left_foot_bone
        self.right_foot = right_foot_bone

        self.left_target = Vector2(0, 0)
        self.right_target = Vector2(0, 0)

        self.step_height = 20.0
        self.step_length = 50.0

        self.phase = 0.0  # 步行相位 0-1
        self.speed = 1.0

    def update(self, delta_time, character_velocity):
        # 更新步行相位
        velocity_length = character_velocity.length()
        if velocity_length > 0.1:
            self.phase += velocity_length * self.speed * delta_time
            self.phase %= 1.0
        else:
            # 静止时逐渐归位
            self.phase = lerp(self.phase, 0.0, 0.1)

        # 计算脚步位置
        self.calculate_foot_positions(character_velocity)

        # 应用IK
        self.solve_foot_ik()

    def calculate_foot_positions(self, velocity):
        # 计算脚步轨迹（椭圆路径）
        left_phase = self.phase
        right_phase = (self.phase + 0.5) % 1.0

        # 前后运动（正弦）
        left_forward = math.sin(left_phase * math.pi * 2) * self.step_length
        right_forward = math.sin(right_phase * math.pi * 2) * self.step_length

        # 上下运动（绝对值正弦）
        left_height = abs(math.sin(left_phase * math.pi)) * self.step_height
        right_height = abs(math.sin(right_phase * math.pi)) * self.step_height

        # 设置目标位置
        direction = velocity.normalize() if velocity.length() > 0 else Vector2(1, 0)

        self.left_target = (direction * left_forward +
                           Vector2(0, -left_height))
        self.right_target = (direction * right_forward +
                            Vector2(0, -right_height))

    def solve_foot_ik(self):
        # 使用两段IK链解决脚步位置
        solve_two_bone_ik(self.left_foot, self.left_target)
        solve_two_bone_ik(self.right_foot, self.right_target)
```

```python
class BroadPhase:
    def __init__(self):
        self.pairs = set()

    def update(self, bodies):
        # 使用AABB快速排除
        self.pairs.clear()

        # 空间分区（如前文的Grid）
        grid = Grid(100, 800, 600)
        for body in bodies:
            grid.add_object(body)

        # 收集潜在碰撞对
        for i, body_a in enumerate(bodies):
            potentials = grid.get_potential_collisions(body_a)

            for body_b in potentials:
                # 确保每对只添加一次
                if id(body_a) < id(body_b):
                    self.pairs.add((body_a, body_b))
                else:
                    self.pairs.add((body_b, body_a))

        return list(self.pairs)
```

#### 时间步长自适应
```python
class AdaptivePhysics:
    def __init__(self, max_step=1/60.0, max_iterations=10):
        self.max_step = max_step
        self.max_iterations = max_iterations

    def update(self, delta_time):
        remaining_time = delta_time
        iterations = 0

        while remaining_time > 0 and iterations < self.max_iterations:
            # 使用固定的最大时间步长
            step_time = min(remaining_time, self.max_step)

            # 执行物理步
            self.physics_step(step_time)

            remaining_time -= step_time
            iterations += 1

        if remaining_time > 0:
            print(f"警告：物理更新未能完全处理 {remaining_time:.3f} 秒")
```

#### 动画LOD（Level of Detail）
```python
class AnimationLOD:
    def __init__(self):
        self.lod_levels = [
            {"update_rate": 1.0, "bone_count": "full"},   # 高：每帧更新，全部骨骼
            {"update_rate": 0.5, "bone_count": "half"},   # 中：每两帧更新，一半骨骼
            {"update_rate": 0.25, "bone_count": "quarter"} # 低：每四帧更新，四分之一骨骼
        ]
        self.current_lod = 0
        self.update_accumulator = 0.0

    def update(self, delta_time, distance_to_camera):
        # 根据距离选择LOD级别
        if distance_to_camera < 10:
            self.current_lod = 0
        elif distance_to_camera < 30:
            self.current_lod = 1
        else:
            self.current_lod = 2

        lod = self.lod_levels[self.current_lod]

        # 累积时间
        self.update_accumulator += delta_time

        # 检查是否需要更新
        if self.update_accumulator >= lod["update_rate"]:
            self.update_accumulator = 0
            return True
        return False
```

```python
class PhysicsDebugRenderer:
    def __init__(self):
        self.shapes = []

    def draw_aabb(self, aabb, color=(255, 0, 0)):
        # 绘制AABB边界框
        self.shapes.append(("rect", aabb.min_x, aabb.min_y,
                           aabb.max_x - aabb.min_x, aabb.max_y - aabb.min_y, color))

    def draw_circle(self, circle, color=(0, 255, 0)):
        # 绘制圆形
        self.shapes.append(("circle", circle.center_x, circle.center_y,
                           circle.radius, color))

    def draw_vector(self, start, vector, color=(0, 0, 255)):
        # 绘制向量（力、速度等）
        end = start + vector
        self.shapes.append(("line", start.x, start.y, end.x, end.y, color))

    def render(self):
        for shape in self.shapes:
            shape_type, *args = shape
            if shape_type == "rect":
                x, y, w, h, color = args
                draw_rect(x, y, w, h, color)
            elif shape_type == "circle":
                x, y, r, color = args
                draw_circle(x, y, r, color)
            elif shape_type == "line":
                x1, y1, x2, y2, color = args
                draw_line(x1, y1, x2, y2, color)

        self.shapes.clear()
```

## 总结

游戏物理与动画编程是创造逼真、响应迅速的游戏体验的关键技术。掌握这些技术需要理解底层原理、实现细节以及性能优化方法。

**关键要点**：
1. **分层实现**：从简单的碰撞检测开始，逐步实现完整的物理模拟
2. **性能优先**：游戏物理和动画是性能敏感区域，需要精心优化
3. **艺术与技术的结合**：动画既是技术也是艺术，需要与美术人员紧密合作
4. **调试工具至关重要**：没有可视化调试，物理和动画问题难以排查

**最佳实践**：
- 使用现有的物理引擎（如Box2D、PhysX）而非从头实现
- 为不同平台和设备调整物理精度
- 实现详尽的调试和可视化工具
- 保持物理和动画系统的模块化和可扩展性

**进阶方向**：
- 软体物理和流体模拟
- 机器学习驱动的动画
- 实时运动匹配（Motion Matching）
- 多角色群体动画

记住：目标是创造让玩家沉浸的游戏体验，而不是追求物理或动画的绝对真实。有时，为了游戏性和性能，需要适当"作弊"和简化。
