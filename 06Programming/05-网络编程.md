#### 自定义应用层协议
```python
class GamePacket:
    PACKET_TYPES = {
        0x01: "CONNECT",
        0x02: "DISCONNECT",
        0x03: "HEARTBEAT",
        0x10: "PLAYER_INPUT",
        0x11: "PLAYER_STATE",
        0x12: "WORLD_STATE",
        0x20: "CHAT_MESSAGE",
        0x30: "GAME_EVENT"
    }

    def __init__(self, packet_type, sequence=0, timestamp=0, payload=b""):
        self.packet_type = packet_type
        self.sequence = sequence  # 包序列号
        self.timestamp = timestamp  # 发送时间戳
        self.payload = payload

    def serialize(self):
        """序列化为字节流"""
        # 头部格式：类型(1) + 序列号(4) + 时间戳(8) + 长度(2) + 载荷
        header = struct.pack(">B I Q H",
                           self.packet_type,
                           self.sequence,
                           self.timestamp,
                           len(self.payload))

        return header + self.payload

    @classmethod
    def deserialize(cls, data):
        """从字节流反序列化"""
        if len(data) < 15:  # 头部最小长度
            return None

        packet_type, sequence, timestamp, payload_len = struct.unpack(">B I Q H", data[:15])

        if len(data) < 15 + payload_len:
            return None

        payload = data[15:15 + payload_len]

        return cls(packet_type, sequence, timestamp, payload)
```

```python
class NetworkManager:
    def __init__(self, use_reliable_udp=True):
        self.socket = None
        self.connected = False
        self.peer_address = None

        # 可靠UDP相关
        self.use_reliable_udp = use_reliable_udp
        self.send_sequence = 0
        self.recv_sequence = 0
        self.ack_bits = 0  # 位字段，确认最近32个包
        self.pending_acks = {}  # 等待确认的包 {sequence: (timestamp, data, retries)}

        # 统计信息
        self.stats = {
            "packets_sent": 0,
            "packets_received": 0,
            "bytes_sent": 0,
            "bytes_received": 0,
            "packet_loss": 0.0,
            "average_rtt": 0.0
        }

    def connect(self, host, port):
        """连接到服务器"""
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_DGRAM if self.use_reliable_udp else socket.SOCK_STREAM)
        self.socket.settimeout(5.0)  # 连接超时

        if self.use_reliable_udp:
            # UDP：无实际连接，记录对方地址
            self.peer_address = (host, port)
            self.connected = True
            self.send_connect_packet()
        else:
            # TCP：建立连接
            self.socket.connect((host, port))
            self.connected = True

    def send_reliable_udp(self, packet_type, payload):
        """发送可靠UDP包"""
        if not self.connected:
            return False

        # 创建包
        packet = GamePacket(packet_type, self.send_sequence, time.time(), payload)
        data = packet.serialize()

        # 保存到等待确认列表
        self.pending_acks[self.send_sequence] = (
            packet.timestamp,
            data,
            0  # 重试次数
        )

        # 发送
        self.socket.sendto(data, self.peer_address)

        # 更新统计
        self.stats["packets_sent"] += 1
        self.stats["bytes_sent"] += len(data)

        self.send_sequence += 1
        return True

    def receive(self):
        """接收数据"""
        try:
            if self.use_reliable_udp:
                data, addr = self.socket.recvfrom(4096)
                self.peer_address = addr  # 可能来自不同端口的响应
            else:
                data = self.socket.recv(4096)

            self.stats["packets_received"] += 1
            self.stats["bytes_received"] += len(data)

            packet = GamePacket.deserialize(data)
            if packet:
                self.handle_packet(packet)

            return packet

        except socket.timeout:
            return None
        except socket.error as e:
            print(f"Socket error: {e}")
            self.disconnect()
            return None

    def handle_packet(self, packet):
        """处理接收到的包"""
        if self.use_reliable_udp:
            # 更新接收序列号
            seq_diff = packet.sequence - self.recv_sequence

            if seq_diff > 0:
                # 新包，更新确认位
                self.ack_bits = (self.ack_bits << 1) | 0x1
                if seq_diff > 1:
                    self.ack_bits |= (1 << (seq_diff - 1))

                self.recv_sequence = packet.sequence

            # 发送确认
            self.send_ack(packet.sequence)

            # 处理包类型
            if packet.packet_type in GamePacket.PACKET_TYPES:
                packet_type_name = GamePacket.PACKET_TYPES[packet.packet_type]
                self.dispatch_packet(packet_type_name, packet)

        else:
            # TCP直接处理
            if packet.packet_type in GamePacket.PACKET_TYPES:
                packet_type_name = GamePacket.PACKET_TYPES[packet.packet_type]
                self.dispatch_packet(packet_type_name, packet)
```

```python
class AuthoritativeServer:
    def __init__(self, tick_rate=30):
        self.tick_rate = tick_rate  # 服务器每秒更新次数
        self.tick_interval = 1.0 / tick_rate
        self.game_state = GameState()
        self.clients = {}  # client_id: ClientInfo

        # 输入缓冲
        self.input_buffer = {}  # client_id: [InputSnapshot]

        # 快照历史（用于调和）
        self.snapshot_history = []  # [(tick, game_state_snapshot)]

    def update(self):
        """服务器主循环"""
        # 1. 收集客户端输入
        inputs = self.collect_inputs()

        # 2. 处理输入（服务器权威验证）
        validated_inputs = self.validate_inputs(inputs)

        # 3. 更新游戏状态
        self.game_state.update(validated_inputs, self.tick_interval)

        # 4. 保存快照
        snapshot = self.game_state.snapshot()
        self.snapshot_history.append((self.current_tick, snapshot))

        # 保留最近100个快照
        if len(self.snapshot_history) > 100:
            self.snapshot_history.pop(0)

        # 5. 广播状态更新
        self.broadcast_state_update()

        self.current_tick += 1

    def collect_inputs(self):
        """收集所有客户端输入"""
        inputs = {}

        for client_id, client in self.clients.items():
            # 获取该客户端的输入（可能多个，用于调和）
            client_inputs = self.input_buffer.get(client_id, [])
            if client_inputs:
                # 使用最新的输入，但保留历史用于调和
                inputs[client_id] = client_inputs[-1]

                # 清除已处理的输入
                self.input_buffer[client_id] = []

        return inputs

    def validate_inputs(self, inputs):
        """验证输入合法性（反作弊）"""
        validated = {}

        for client_id, input_snapshot in inputs.items():
            # 基本验证
            if self.validate_input_snapshot(client_id, input_snapshot):
                # 物理验证（如移动速度限制）
                if self.physics_validation(client_id, input_snapshot):
                    validated[client_id] = input_snapshot
                else:
                    print(f"物理验证失败: client {client_id}")
                    # 可能的作弊行为，记录或断开连接
            else:
                print(f"输入验证失败: client {client_id}")

        return validated

    def broadcast_state_update(self):
        """广播游戏状态给所有客户端"""
        # 创建状态更新包
        state_update = self.create_state_update()

        for client_id, client in self.clients.items():
            # 为每个客户端定制更新（只发送相关信息）
            client_update = self.filter_state_for_client(state_update, client_id)

            # 发送
            self.send_to_client(client_id, client_update)
```

```python
class ClientPrediction:
    def __init__(self, server):
        self.server = server
        self.local_state = LocalGameState()

        # 输入历史
        self.input_history = []  # [(tick, input, local_state_snapshot)]

        # 预测状态
        self.predicted_states = {}  # tick: predicted_state

    def apply_input(self, input_snapshot):
        """应用输入并进行本地预测"""
        # 1. 保存当前状态快照
        current_snapshot = self.local_state.snapshot()

        # 2. 应用输入
        self.local_state.apply_input(input_snapshot)

        # 3. 记录到历史
        self.input_history.append((
            self.server.current_tick,
            input_snapshot,
            current_snapshot
        ))

        # 4. 预测未来状态
        predicted_tick = self.server.current_tick + 1
        self.predict_state(predicted_tick)

        # 5. 发送输入到服务器
        self.server.send_input(input_snapshot)

    def reconcile(self, server_state, server_tick):
        """服务器调和：修正预测错误"""
        # 1. 找到服务器确认的最后一个输入
        confirmed_inputs = []

        for tick, input_snap, local_snap in self.input_history:
            if tick <= server_tick:
                confirmed_inputs.append((tick, input_snap))

        # 2. 回滚到最后一个确认状态
        if confirmed_inputs:
            last_confirmed_tick, _ = confirmed_inputs[-1]

            # 找到对应的本地快照
            for tick, input_snap, local_snap in self.input_history:
                if tick == last_confirmed_tick:
                    # 回滚到这个状态
                    self.local_state.restore(local_snap)
                    break

        # 3. 清除已确认的历史
        self.input_history = [
            (t, i, s) for t, i, s in self.input_history
            if t > server_tick
        ]

        # 4. 从服务器状态重新应用输入
        self.local_state.restore(server_state)

        for tick, input_snap in confirmed_inputs:
            if tick > server_tick:  # 重新应用未确认的输入
                self.local_state.apply_input(input_snap)

        # 5. 重新预测
        self.repredict_future()
```

```python
class PeerToPeerNetwork:
    def __init__(self, is_host=False):
        self.is_host = is_host
        self.peers = {}  # peer_id: PeerInfo
        self.game_state = DistributedGameState()

        # 锁步同步（用于RTS等确定性游戏）
        self.lockstep_turn = 0
        self.pending_inputs = {}  # turn: {peer_id: input}
        self.confirmed_inputs = {}  # turn: [peer_id]

    def lockstep_update(self):
        """锁步更新：所有客户端同步执行"""
        if self.is_host:
            # 主机收集并广播输入
            self.host_lockstep()
        else:
            # 客户端发送输入并等待
            self.client_lockstep()

    def host_lockstep(self):
        """主机锁步逻辑"""
        current_turn = self.lockstep_turn

        # 1. 收集所有客户端输入
        all_inputs_received = self.collect_turn_inputs(current_turn)

        if all_inputs_received:
            # 2. 广播输入集合给所有客户端
            self.broadcast_turn_inputs(current_turn, self.pending_inputs[current_turn])

            # 3. 应用输入并更新游戏状态
            self.apply_turn_inputs(current_turn)

            # 4. 前进到下一回合
            self.lockstep_turn += 1

            # 5. 清除已处理的输入
            if current_turn in self.pending_inputs:
                del self.pending_inputs[current_turn]

    def client_lockstep(self):
        """客户端锁步逻辑"""
        current_turn = self.lockstep_turn

        # 1. 发送本回合输入到主机
        self.send_input_to_host(current_turn, self.get_local_input())

        # 2. 等待主机广播的所有输入
        turn_inputs = self.wait_for_turn_inputs(current_turn)

        if turn_inputs:
            # 3. 应用输入（与主机完全相同的顺序）
            self.apply_turn_inputs(current_turn, turn_inputs)

            # 4. 前进到下一回合
            self.lockstep_turn += 1

    def apply_turn_inputs(self, turn, inputs=None):
        """应用回合输入（确定性执行）"""
        if inputs is None:
            inputs = self.pending_inputs.get(turn, {})

        # 按确定的顺序应用输入（如按peer_id排序）
        sorted_peer_ids = sorted(inputs.keys())

        for peer_id in sorted_peer_ids:
            input_snapshot = inputs[peer_id]
            self.game_state.apply_input(peer_id, input_snapshot)
```

```python
class StateSynchronization:
    def __init__(self, sync_rate=10):  # 每秒同步10次
        self.sync_rate = sync_rate
        self.sync_interval = 1.0 / sync_rate
        self.last_sync_time = 0

        # 状态压缩
        self.state_encoder = StateEncoder()
        self.delta_encoder = DeltaEncoder()

        # 优先级系统
        self.update_priorities = {
            "player": 1.0,
            "nearby_enemy": 0.8,
            "far_enemy": 0.4,
            "environment": 0.2
        }

    def sync_state(self, current_time, full_state, client_info):
        """同步状态给客户端"""
        # 检查是否到同步时间
        if current_time - self.last_sync_time < self.sync_interval:
            return None

        self.last_sync_time = current_time

        # 1. 根据客户端位置过滤状态
        filtered_state = self.filter_state_by_distance(full_state, client_info.position)

        # 2. 应用优先级
        prioritized_state = self.apply_priorities(filtered_state, client_info)

        # 3. 增量编码
        if client_info.last_state:
            # 发送差异
            delta = self.delta_encoder.encode(client_info.last_state, prioritized_state)

            # 压缩
            compressed = self.compress_delta(delta)

            # 更新客户端最后状态
            client_info.last_state = prioritized_state

            return compressed
        else:
            # 第一次发送完整状态
            encoded = self.state_encoder.encode(prioritized_state)
            compressed = self.compress_state(encoded)

            client_info.last_state = prioritized_state

            return compressed

    def filter_state_by_distance(self, full_state, client_position):
        """根据距离过滤状态"""
        filtered = {}

        for entity_id, entity_state in full_state.items():
            distance = self.calculate_distance(entity_state.position, client_position)

            # 根据实体类型和距离决定是否包含
            if self.should_include_entity(entity_state, distance):
                # 根据距离决定细节级别
                detail_level = self.get_detail_level(distance)
                simplified_state = self.simplify_state(entity_state, detail_level)

                filtered[entity_id] = simplified_state

        return filtered

    def apply_priorities(self, state, client_info):
        """应用更新优先级"""
        prioritized = {}

        for entity_id, entity_state in state.items():
            # 计算实体对客户端的优先级
            priority = self.calculate_priority(entity_state, client_info)

            # 根据优先级决定更新频率
            if self.should_update_now(priority, entity_id, client_info):
                prioritized[entity_id] = {
                    "state": entity_state,
                    "priority": priority,
                    "last_update": time.time()
                }

        return prioritized

    def calculate_priority(self, entity_state, client_info):
        """计算实体更新优先级"""
        base_priority = self.update_priorities.get(entity_state.type, 0.5)

        # 距离因子
        distance = self.calculate_distance(entity_state.position, client_info.position)
        distance_factor = max(0, 1.0 - distance / 100.0)  # 100单位外优先级为0

        # 重要性因子（如玩家正在交互的实体）
        importance_factor = 1.0
        if entity_id in client_info.interacting_with:
            importance_factor = 2.0

        # 变化率因子（移动快的实体需要更频繁更新）
        velocity_factor = 1.0 + min(1.0, entity_state.velocity.length() / 10.0)

        return base_priority * distance_factor * importance_factor * velocity_factor
```

### 3.2 实体插值

```python
class EntityInterpolation:
    def __init__(self, interpolation_time=0.1):  # 100ms插值延迟
        self.interpolation_time = interpolation_time
        self.entity_states = {}  # entity_id: [ (time, state), ... ]

    def add_state_update(self, entity_id, timestamp, state):
        """添加状态更新"""
        if entity_id not in self.entity_states:
            self.entity_states[entity_id] = []

        states = self.entity_states[entity_id]

        # 添加新状态
        states.append((timestamp, state))

        # 保持按时间排序
        states.sort(key=lambda x: x[0])

        # 移除旧状态（比当前时间早interpolation_time * 2）
        cutoff_time = timestamp - self.interpolation_time * 2
        self.entity_states[entity_id] = [
            (t, s) for t, s in states if t >= cutoff_time
        ]

    def get_interpolated_state(self, entity_id, current_time):
        """获取插值后的状态"""
        if entity_id not in self.entity_states:
            return None

        states = self.entity_states[entity_id]

        if len(states) < 2:
            # 没有足够数据，返回最新状态
            return states[-1][1] if states else None

        # 计算插值目标时间
        target_time = current_time - self.interpolation_time

        # 找到包围目标时间的两个状态
        prev_state = None
        next_state = None

        for i in range(len(states) - 1):
            t1, s1 = states[i]
            t2, s2 = states[i + 1]

            if t1 <= target_time <= t2:
                prev_state = (t1, s1)
                next_state = (t2, s2)
                break

        if not prev_state or not next_state:
            # 目标时间在范围外，返回最近的状态
            return states[-1][1]

        # 计算插值因子
        t1, s1 = prev_state
        t2, s2 = next_state

        if t2 == t1:  # 避免除零
            return s1

        alpha = (target_time - t1) / (t2 - t1)

        # 线性插值
        interpolated = self.interpolate_states(s1, s2, alpha)

        return interpolated

    def interpolate_states(self, state1, state2, alpha):
        """插值两个状态"""
        interpolated = {}

        # 插值位置
        if "position" in state1 and "position" in state2:
            pos1 = state1["position"]
            pos2 = state2["position"]
            interpolated["position"] = (
                pos1[0] + (pos2[0] - pos1[0]) * alpha,
                pos1[1] + (pos2[1] - pos1[1]) * alpha
            )

        # 插值旋转（角度）
        if "rotation" in state1 and "rotation" in state2:
            rot1 = state1["rotation"]
            rot2 = state2["rotation"]

            # 处理角度环绕
            diff = rot2 - rot1
            if diff > 180:
                diff -= 360
            elif diff < -180:
                diff += 360

            interpolated["rotation"] = rot1 + diff * alpha

        # 其他属性使用最新值或插值
        for key in state2:
            if key not in ["position", "rotation"]:
                interpolated[key] = state2[key]

        return interpolated
```

```python
class BandwidthOptimizer:
    def __init__(self):
        self.compression = {
            "position": DeltaCompression(),
            "rotation": AngleCompression(),
            "state_flags": BitfieldCompression()
        }

        self.prediction_models = {}  # entity_id: PredictionModel

    def compress_update(self, entity_id, old_state, new_state):
        """压缩状态更新"""
        compressed = {}

        for key, compressor in self.compression.items():
            if key in new_state:
                if key in old_state:
                    # 增量压缩
                    compressed[key] = compressor.compress_delta(old_state[key], new_state[key])
                else:
                    # 完整压缩
                    compressed[key] = compressor.compress_full(new_state[key])

        # 应用预测模型
        if entity_id in self.prediction_models:
            model = self.prediction_models[entity_id]
            predicted = model.predict(old_state)

            # 只发送与预测的差异
            diff = self.calculate_difference(predicted, new_state)

            if self.is_significant_difference(diff):
                compressed["diff"] = diff
            else:
                # 差异太小，不发送
                return None

        return compressed

    def adaptive_update_rate(self, client_info):
        """自适应更新频率"""
        # 基于连接质量调整
        connection_quality = client_info.connection_quality

        if connection_quality > 0.8:  # 优秀
            update_rate = 30  # Hz
        elif connection_quality > 0.5:  # 良好
            update_rate = 20  # Hz
        elif connection_quality > 0.3:  # 一般
            update_rate = 10  # Hz
        else:  # 差
            update_rate = 5   # Hz

        # 基于重要性调整
        importance = client_info.session_importance

        if importance > 0.7:  # 重要玩家（如比赛主持人）
            update_rate = min(update_rate * 1.5, 60)
        elif importance < 0.3:  # 不活跃玩家
            update_rate = max(update_rate * 0.5, 1)

        return update_rate

    def priority_based_scheduling(self, updates, available_bandwidth):
        """基于优先级的更新调度"""
        # 计算每个更新的优先级分数
        scored_updates = []

        for update in updates:
            score = self.calculate_update_score(update)
            size = self.estimate_update_size(update)
            scored_updates.append((score, size, update))

        # 按分数降序排序
        scored_updates.sort(key=lambda x: x[0], reverse=True)

        # 选择能放入带宽的更新
        selected = []
        total_size = 0

        for score, size, update in scored_updates:
            if total_size + size <= available_bandwidth:
                selected.append(update)
                total_size += size
            else:
                break

        return selected
```

```python
class LagCompensation:
    def __init__(self, max_lag_time=0.5):  # 最大补偿500ms
        self.max_lag_time = max_lag_time
        self.player_states = {}  # player_id: [(timestamp, state)]

    def record_player_state(self, player_id, timestamp, state):
        """记录玩家状态历史"""
        if player_id not in self.player_states:
            self.player_states[player_id] = []

        history = self.player_states[player_id]
        history.append((timestamp, state))

        # 保持历史排序，移除旧状态
        history.sort(key=lambda x: x[0])
        cutoff = timestamp - self.max_lag_time
        self.player_states[player_id] = [(t, s) for t, s in history if t >= cutoff]

    def rewind_time(self, shooter_id, shot_time, target_id):
        """时间倒流：计算射击时刻的目标位置"""
        if target_id not in self.player_states:
            return None

        history = self.player_states[target_id]

        # 找到射击时间前后的状态
        prev_state = None
        next_state = None

        for i in range(len(history) - 1):
            t1, s1 = history[i]
            t2, s2 = history[i + 1]

            if t1 <= shot_time <= t2:
                prev_state = (t1, s1)
                next_state = (t2, s2)
                break

        if not prev_state:
            # 射击时间在记录范围外，使用最近状态
            return history[-1][1] if history else None

        # 插值
        t1, s1 = prev_state
        t2, s2 = next_state
        alpha = (shot_time - t1) / (t2 - t1) if t2 != t1 else 0

        rewound_state = self.interpolate_states(s1, s2, alpha)
        return rewound_state

    def process_shot(self, shooter_id, shot_time, target_id, hit_position):
        """处理射击（带延迟补偿）"""
        # 1. 倒流到射击时间
        target_state = self.rewind_time(shooter_id, shot_time, target_id)

        if not target_state:
            return False

        # 2. 使用倒流后的状态进行命中检测
        hit = self.check_hit(target_state, hit_position)

        # 3. 如果命中，应用伤害（使用当前时间）
        if hit:
            damage = self.calculate_damage(shooter_id, target_id)
            self.apply_damage(target_id, damage, time.time())

            # 记录命中（用于反作弊验证）
            self.log_hit(shooter_id, target_id, shot_time, hit_position)

        return hit
```

```python
class NetworkSecurity:
    def __init__(self):
        self.encryption = AESCipher()
        self.checksum = CRC32()

        # 反作弊系统
        self.cheat_detection = CheatDetectionSystem()

        # 会话管理
        self.sessions = {}  # session_id: SessionInfo

    def secure_send(self, data, session_id):
        """安全发送数据"""
        if session_id not in self.sessions:
            raise ValueError("Invalid session")

        session = self.sessions[session_id]

        # 1. 添加序列号（防重放攻击）
        sequence = session.next_sequence()
        header = struct.pack(">I", sequence)

        # 2. 计算校验和
        checksum = self.checksum.calculate(data)

        # 3. 加密
        encrypted = self.encryption.encrypt(header + struct.pack(">I", checksum) + data)

        # 4. 发送
        return encrypted

    def secure_receive(self, encrypted_data, session_id):
        """安全接收数据"""
        if session_id not in self.sessions:
            raise ValueError("Invalid session")

        session = self.sessions[session_id]

        try:
            # 1. 解密
            decrypted = self.encryption.decrypt(encrypted_data)

            # 2. 解析
            sequence = struct.unpack(">I", decrypted[:4])[0]
            checksum = struct.unpack(">I", decrypted[4:8])[0]
            data = decrypted[8:]

            # 3. 检查序列号（防重放）
            if not session.validate_sequence(sequence):
                print(f"Replay attack detected: sequence {sequence}")
                return None

            # 4. 验证校验和
            if self.checksum.calculate(data) != checksum:
                print("Checksum mismatch")
                return None

            # 5. 反作弊检查
            if self.cheat_detection.check_packet(data, session_id):
                print("Cheat detected in packet")
                return None

            return data

        except Exception as e:
            print(f"Security error: {e}")
            return None

    def validate_client_input(self, client_id, input_snapshot):
        """验证客户端输入"""
        # 1. 范围检查
        if not self.check_input_range(input_snapshot):
            return False

        # 2. 速率限制
        if not self.check_input_rate(client_id, input_snapshot.timestamp):
            return False

        # 3. 物理可能性检查
        if not self.check_physics_possibility(client_id, input_snapshot):
            return False

        # 4. 行为模式分析
        if self.cheat_detection.analyze_behavior(client_id, input_snapshot):
            return False

        return True
```

```python
class ServerAuthority:
    def __init__(self):
        self.client_states = {}  # client_id: ClientState
        self.validators = {
            "movement": MovementValidator(),
            "combat": CombatValidator(),
            "inventory": InventoryValidator()
        }

    def validate_action(self, client_id, action_type, action_data):
        """验证客户端动作"""
        client_state = self.client_states.get(client_id)

        if not client_state:
            return False

        # 获取对应的验证器
        validator = self.validators.get(action_type)

        if not validator:
            print(f"No validator for action type: {action_type}")
            return False

        # 验证动作
        valid, new_state = validator.validate(client_state, action_data)

        if valid:
            # 更新客户端状态
            self.client_states[client_id] = new_state

            # 记录验证通过的动作（用于调试和反作弊）
            self.log_validated_action(client_id, action_type, action_data)

            return True
        else:
            # 记录可疑动作
            self.log_suspicious_action(client_id, action_type, action_data)

            # 如果多次验证失败，标记为可疑
            self.track_validation_failures(client_id)

            return False

    class MovementValidator:
        def validate(self, client_state, move_data):
            """验证移动动作"""
            # 1. 检查移动速度是否可能
            current_pos = client_state.position
            target_pos = move_data["position"]
            time_delta = move_data["time_delta"]

            distance = self.calculate_distance(current_pos, target_pos)
            speed = distance / time_delta

            max_speed = client_state.max_speed

            # 考虑网络延迟和预测误差
            allowed_speed = max_speed * 1.2  # 允许20%的超速

            if speed > allowed_speed:
                return False, client_state

            # 2. 检查移动路径是否合法
            if not self.is_valid_path(current_pos, target_pos, client_state.collision_map):
                return False, client_state

            # 3. 更新状态
            new_state = client_state.copy()
            new_state.position = target_pos
            new_state.last_move_time = move_data["timestamp"]

            return True, new_state

    class CombatValidator:
        def validate(self, client_state, combat_data):
            """验证战斗动作"""
            # 1. 检查攻击冷却
            current_time = combat_data["timestamp"]
            time_since_last_attack = current_time - client_state.last_attack_time

            if time_since_last_attack < client_state.attack_cooldown:
                return False, client_state

            # 2. 检查攻击范围
            target_id = combat_data["target_id"]
            target_state = self.get_target_state(target_id)

            if not target_state:
                return False, client_state

            distance = self.calculate_distance(client_state.position, target_state.position)

            if distance > client_state.attack_range:
                return False, client_state

            # 3. 检查视线
            if not self.has_line_of_sight(client_state.position, target_state.position):
                return False, client_state

            # 4. 更新状态
            new_state = client_state.copy()
            new_state.last_attack_time = current_time

            return True, new_state
```

```python
class NetworkMonitor:
    def __init__(self):
        self.metrics = {
            "latency": CircularBuffer(100),
            "packet_loss": CircularBuffer(100),
            "bandwidth": CircularBuffer(100),
            "jitter": CircularBuffer(100)
        }

        self.alerts = []
        self.thresholds = {
            "high_latency": 200,  # ms
            "high_packet_loss": 0.1,  # 10%
            "high_jitter": 50  # ms
        }

    def update_metrics(self, packet_info):
        """更新网络指标"""
        # 计算延迟
        if packet_info.sent_time and packet_info.received_time:
            latency = (packet_info.received_time - packet_info.sent_time) * 1000  # ms
            self.metrics["latency"].add(latency)

        # 计算丢包率
        if packet_info.sequence_gap > 0:
            loss_rate = packet_info.sequence_gap / packet_info.expected_sequence
            self.metrics["packet_loss"].add(loss_rate)

        # 计算带宽
        if packet_info.size and packet_info.delta_time:
            bandwidth = packet_info.size / packet_info.delta_time  # bytes/sec
            self.metrics["bandwidth"].add(bandwidth)

        # 计算抖动（延迟变化）
        if len(self.metrics["latency"]) >= 2:
            last_latency = self.metrics["latency"][-2]
            current_latency = self.metrics["latency"][-1]
            jitter = abs(current_latency - last_latency)
            self.metrics["jitter"].add(jitter)

        # 检查阈值
        self.check_thresholds()

    def check_thresholds(self):
        """检查是否超过阈值"""
        avg_latency = self.metrics["latency"].average()
        avg_packet_loss = self.metrics["packet_loss"].average()
        avg_jitter = self.metrics["jitter"].average()

        if avg_latency > self.thresholds["high_latency"]:
            self.alert("high_latency", f"High latency: {avg_latency:.1f}ms")

        if avg_packet_loss > self.thresholds["high_packet_loss"]:
            self.alert("high_packet_loss", f"High packet loss: {avg_packet_loss:.1%}")

        if avg_jitter > self.thresholds["high_jitter"]:
            self.alert("high_jitter", f"High jitter: {avg_jitter:.1f}ms")

    def get_diagnostics(self):
        """获取网络诊断信息"""
        return {
            "latency": {
                "current": self.metrics["latency"].latest(),
                "average": self.metrics["latency"].average(),
                "min": self.metrics["latency"].min(),
                "max": self.metrics["latency"].max()
            },
            "packet_loss": {
                "current": self.metrics["packet_loss"].latest(),
                "average": self.metrics["packet_loss"].average()
            },
            "bandwidth": {
                "current": self.metrics["bandwidth"].latest(),
                "average": self.metrics["bandwidth"].average()
            },
            "jitter": {
                "current": self.metrics["jitter"].latest(),
                "average": self.metrics["jitter"].average()
            },
            "alerts": self.alerts[-10:]  # 最近10个警报
        }
```

## 总结

游戏网络编程是创建多人游戏体验的核心技术，涉及从底层协议到高级同步算法的多个层面。掌握这些技术需要理解网络特性、游戏设计和性能优化的平衡。

**关键要点**：
1. **选择合适的架构**：根据游戏类型选择客户端-服务器或点对点架构
2. **处理网络不确定性**：延迟、丢包和乱序是常态，需要相应设计
3. **安全性至关重要**：网络游戏容易受到作弊攻击，需要多层防护
4. **性能优化持续进行**：网络代码需要不断监控和优化

**最佳实践**：
- 使用现有的网络库和中间件（如Photon、Mirror、Netcode for GameObjects）
- 实现详细的网络调试和可视化工具
- 进行大规模压力测试，模拟真实网络条件
- 设计时考虑扩展性，支持更多玩家和复杂交互

**进阶方向**：
- 专用服务器和云游戏架构
- 区域服务器和全球网络优化
- 预测性网络和AI驱动的网络优化
- WebRTC和P2P网络游戏

记住：网络游戏的最终目标是提供流畅、公平、有趣的社交体验。技术应该服务于这个目标，而不是成为障碍。
