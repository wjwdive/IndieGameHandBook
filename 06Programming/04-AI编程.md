## 1、AI基础：有限状态机

### 1.1 有限状态机（FSM）基础

有限状态机是游戏AI最基础、最常用的技术，特别适合行为模式固定的AI角色。

```python
class FSMState:
    def enter(self, entity):
        """进入状态时调用"""
        pass

    def exit(self, entity):
        """离开状态时调用"""
        pass

    def update(self, entity, delta_time):
        """每帧更新"""
        pass

    def check_transitions(self, entity):
        """检查状态转换条件，返回新状态或None"""
        return None

class IdleState(FSMState):
    def enter(self, entity):
        entity.animation.play("idle")
        entity.idle_timer = 0

    def update(self, entity, delta_time):
        entity.idle_timer += delta_time

        # 随机移动
        if entity.idle_timer > 2.0 and random.random() < 0.1:
            entity.target_position = entity.get_random_position()

    def check_transitions(self, entity):
        # 检查是否看到玩家
        if entity.can_see_player():
            return ChaseState()

        # 检查是否受到伤害
        if entity.health < entity.last_health:
            return FleeState()

        # 检查是否有目标位置
        if hasattr(entity, 'target_position'):
            return MoveToState()

        return None

class ChaseState(FSMState):
    def enter(self, entity):
        entity.animation.play("run")
        entity.speed = entity.chase_speed

    def update(self, entity, delta_time):
        # 向玩家移动
        player_pos = entity.get_player_position()
        direction = (player_pos - entity.position).normalize()
        entity.position += direction * entity.speed * delta_time

    def check_transitions(self, entity):
        # 检查玩家是否在攻击范围内
        if entity.distance_to_player() < entity.attack_range:
            return AttackState()

        # 检查是否丢失玩家
        if not entity.can_see_player():
            return SearchState()

        return None
```

### 1.2 分层状态机（HFSM）

当状态机变得复杂时，可以使用分层状态机管理：

```python
class HierarchicalFSM:
    def __init__(self):
        self.current_state = None
        self.parent_state = None
        self.stack = []  # 状态栈，用于嵌套状态

    def push_state(self, state):
        """压入新状态"""
        if self.current_state:
            self.current_state.exit()
            self.stack.append(self.current_state)

        self.current_state = state
        self.current_state.enter()

    def pop_state(self):
        """弹出当前状态，返回上一状态"""
        if self.current_state:
            self.current_state.exit()

        if self.stack:
            self.current_state = self.stack.pop()
            self.current_state.enter()
        else:
            self.current_state = None

    def update(self, delta_time):
        """更新当前状态"""
        if self.current_state:
            self.current_state.update(delta_time)

            # 检查状态转换
            new_state = self.current_state.check_transitions()
            if new_state:
                self.change_state(new_state)

    def change_state(self, new_state):
        """切换到新状态"""
        if self.current_state:
            self.current_state.exit()

        self.current_state = new_state
        self.current_state.enter()
```

## 2、行为树（Behavior Tree）

### 2.1 行为树基础

行为树是比状态机更灵活、更模块化的AI架构：

```python
class BTNode:
    """行为树节点基类"""
    def __init__(self):
        self.children = []
        self.parent = None

    def add_child(self, child):
        self.children.append(child)
        child.parent = self

    def execute(self, entity):
        """执行节点，返回状态：SUCCESS, FAILURE, RUNNING"""
        raise NotImplementedError

class Sequence(BTNode):
    """顺序节点：所有子节点成功才成功，一个失败就失败"""
    def execute(self, entity):
        for child in self.children:
            result = child.execute(entity)
            if result == "FAILURE":
                return "FAILURE"
            if result == "RUNNING":
                return "RUNNING"
        return "SUCCESS"

class Selector(BTNode):
    """选择节点：一个子节点成功就成功，所有失败才失败"""
    def execute(self, entity):
        for child in self.children:
            result = child.execute(entity)
            if result == "SUCCESS":
                return "SUCCESS"
            if result == "RUNNING":
                return "RUNNING"
        return "FAILURE"

class Parallel(BTNode):
    """并行节点：同时执行所有子节点"""
    def __init__(self, success_policy="ALL", failure_policy="ONE"):
        super().__init__()
        self.success_policy = success_policy  # ALL, ONE
        self.failure_policy = failure_policy  # ALL, ONE

    def execute(self, entity):
        successes = 0
        failures = 0
        running = 0

        for child in self.children:
            result = child.execute(entity)
            if result == "SUCCESS":
                successes += 1
            elif result == "FAILURE":
                failures += 1
            else:  # RUNNING
                running += 1

        # 根据策略决定结果
        if self.success_policy == "ALL" and successes == len(self.children):
            return "SUCCESS"
        if self.success_policy == "ONE" and successes > 0:
            return "SUCCESS"
        if self.failure_policy == "ALL" and failures == len(self.children):
            return "FAILURE"
        if self.failure_policy == "ONE" and failures > 0:
            return "FAILURE"

        return "RUNNING"
```

### 2.2 行为节点

```python
class Condition(BTNode):
    """条件节点：检查条件"""
    def __init__(self, condition_func):
        super().__init__()
        self.condition_func = condition_func

    def execute(self, entity):
        if self.condition_func(entity):
            return "SUCCESS"
        return "FAILURE"

class Action(BTNode):
    """动作节点：执行动作"""
    def __init__(self, action_func):
        super().__init__()
        self.action_func = action_func

    def execute(self, entity):
        return self.action_func(entity)

class Decorator(BTNode):
    """装饰器节点：修改子节点行为"""
    def __init__(self, child=None):
        super().__init__()
        if child:
            self.add_child(child)

class Repeat(Decorator):
    """重复执行子节点"""
    def __init__(self, child, times=0):  # 0表示无限重复
        super().__init__(child)
        self.times = times
        self.count = 0

    def execute(self, entity):
        if self.times > 0 and self.count >= self.times:
            return "SUCCESS"

        result = self.children[0].execute(entity)
        if result in ["SUCCESS", "FAILURE"]:
            self.count += 1

            if self.times > 0 and self.count >= self.times:
                return "SUCCESS"

            # 重置子节点（如果需要）
            if hasattr(self.children[0], 'reset'):
                self.children[0].reset()

        return "RUNNING"

class Inverter(Decorator):
    """反转子节点结果"""
    def execute(self, entity):
        result = self.children[0].execute(entity)
        if result == "SUCCESS":
            return "FAILURE"
        elif result == "FAILURE":
            return "SUCCESS"
        return result
```

```python
def build_guard_behavior_tree():
    """构建守卫AI的行为树"""
    root = Selector()

    # 优先级1：战斗
    combat_sequence = Sequence()
    combat_sequence.add_child(Condition(lambda e: e.can_see_player()))
    combat_sequence.add_child(Condition(lambda e: e.distance_to_player() < e.attack_range))
    combat_sequence.add_child(Action(lambda e: e.attack_player()))

    # 优先级2：追击
    chase_sequence = Sequence()
    chase_sequence.add_child(Condition(lambda e: e.can_see_player()))
    chase_sequence.add_child(Action(lambda e: e.move_to_player()))
    root.add_child(chase_sequence)

    # 优先级3：搜索
    search_selector = Selector()

    # 选项3.1：调查最后已知位置
    investigate_sequence = Sequence()
    investigate_sequence.add_child(Condition(lambda e: e.has_last_known_player_pos()))
    investigate_sequence.add_child(Action(lambda e: e.move_to_last_known_pos()))
    investigate_sequence.add_child(Action(lambda e: e.search_area()))
    search_selector.add_child(investigate_sequence)

    # 选项3.2：随机巡逻
    patrol_sequence = Sequence()
    patrol_sequence.add_child(Action(lambda e: e.patrol_random_point()))
    patrol_sequence.add_child(Action(lambda e: e.idle_for_time(3.0)))
    search_selector.add_child(patrol_sequence)

    root.add_child(search_selector)

    # 优先级4：闲逛
    idle_sequence = Sequence()
    idle_sequence.add_child(Action(lambda e: e.idle_for_time(5.0)))
    idle_sequence.add_child(Action(lambda e: e.move_to_random_point()))
    root.add_child(idle_sequence)

    return root
```

## 3、路径寻找与导航

### 3.1 A*算法实现

```python
class AStar:
    def __init__(self, grid_width, grid_height):
        self.width = grid_width
        self.height = grid_height
        self.grid = [[0 for _ in range(grid_width)] for _ in range(grid_height)]

    class Node:
        def __init__(self, x, y):
            self.x = x
            self.y = y
            self.g = 0  # 从起点到当前节点的成本
            self.h = 0  # 启发式成本（到终点的估计）
            self.f = 0  # g + h
            self.parent = None

        def __lt__(self, other):
            return self.f < other.f

    def find_path(self, start_x, start_y, end_x, end_y):
        """查找从起点到终点的路径"""
        # 创建起点和终点节点
        start_node = self.Node(start_x, start_y)
        end_node = self.Node(end_x, end_y)

        # 开放列表和关闭列表
        open_list = []
        closed_set = set()

        # 添加起点到开放列表
        heapq.heappush(open_list, start_node)

        while open_list:
            # 获取F值最小的节点
            current_node = heapq.heappop(open_list)

            # 到达终点
            if current_node.x == end_node.x and current_node.y == end_node.y:
                return self.reconstruct_path(current_node)

            # 添加到关闭列表
            closed_set.add((current_node.x, current_node.y))

            # 检查邻居
            for neighbor in self.get_neighbors(current_node):
                neighbor_pos = (neighbor.x, neighbor.y)

                # 如果邻居在关闭列表中或不可通行，跳过
                if neighbor_pos in closed_set or not self.is_walkable(neighbor.x, neighbor.y):
                    continue

                # 计算新的G值
                tentative_g = current_node.g + self.get_cost(current_node, neighbor)

                # 如果邻居不在开放列表中，或新的G值更小
                in_open = any(n.x == neighbor.x and n.y == neighbor.y for n in open_list)
                if not in_open or tentative_g < neighbor.g:
                    neighbor.g = tentative_g
                    neighbor.h = self.heuristic(neighbor, end_node)
                    neighbor.f = neighbor.g + neighbor.h
                    neighbor.parent = current_node

                    if not in_open:
                        heapq.heappush(open_list, neighbor)

        # 未找到路径
        return []

    def get_neighbors(self, node):
        """获取节点的邻居（8方向）"""
        neighbors = []
        for dx in [-1, 0, 1]:
            for dy in [-1, 0, 1]:
                if dx == 0 and dy == 0:
                    continue

                x, y = node.x + dx, node.y + dy
                if 0 <= x < self.width and 0 <= y < self.height:
                    neighbors.append(self.Node(x, y))

        return neighbors

    def heuristic(self, a, b):
        """启发式函数（曼哈顿距离）"""
        return abs(a.x - b.x) + abs(a.y - b.y)

    def get_cost(self, a, b):
        """计算从a到b的成本"""
        dx = abs(a.x - b.x)
        dy = abs(a.y - b.y)
        if dx == 1 and dy == 1:
            return 1.414  # 对角线成本
        return 1.0  # 直线成本

    def reconstruct_path(self, node):
        """从终点重构路径"""
        path = []
        current = node
        while current:
            path.append((current.x, current.y))
            current = current.parent
        return list(reversed(path))
```

### 3.2 导航网格（NavMesh）

```python
class NavMesh:
    def __init__(self):
        self.polygons = []  # 凸多边形列表
        self.graph = None   # 导航图

    def build_from_level(self, level_geometry):
        """从关卡几何构建导航网格"""
        # 1. 将关卡几何分解为凸多边形
        self.polygons = self.decompose_to_convex(level_geometry)

        # 2. 构建导航图
        self.build_navigation_graph()

    def build_navigation_graph(self):
        """构建多边形之间的导航图"""
        self.graph = {}

        for i, poly_a in enumerate(self.polygons):
            self.graph[i] = []

            for j, poly_b in enumerate(self.polygons):
                if i == j:
                    continue

                # 检查多边形是否相邻
                if self.are_polygons_adjacent(poly_a, poly_b):
                    # 计算连接点（共享边的中点）
                    connection_point = self.get_connection_point(poly_a, poly_b)

                    # 计算成本（距离）
                    cost = self.calculate_travel_cost(poly_a.center, poly_b.center)

                    self.graph[i].append({
                        "to": j,
                        "point": connection_point,
                        "cost": cost
                    })

    def find_path(self, start_pos, end_pos):
        """在导航网格上查找路径"""
        # 1. 找到起点和终点所在的多边形
        start_poly = self.find_polygon_containing(start_pos)
        end_poly = self.find_polygon_containing(end_pos)

        if not start_poly or not end_poly:
            return None

        # 2. 使用A*在导航图上查找多边形路径
        poly_path = self.astar_on_graph(start_poly, end_poly)

        if not poly_path:
            return None

        # 3. 将多边形路径转换为点路径
        point_path = self.convert_poly_path_to_points(poly_path, start_pos, end_pos)

        # 4. 路径平滑（Funnel算法）
        smooth_path = self.funnel_algorithm(point_path)

        return smooth_path

    def funnel_algorithm(self, portals):
        """漏斗算法，用于在通道中寻找最短路径"""
        if len(portals) < 2:
            return portals

        # 初始化漏斗
        apex = portals[0][0]  # 起点
        left = portals[0][0]  # 左边界
        right = portals[0][1]  # 右边界

        path = [apex]

        for i in range(1, len(portals)):
            # 更新左右边界
            new_left = portals[i][0]
            new_right = portals[i][1]

            # 检查是否需要更新顶点
            if self.is_right_of(apex, left, new_left):
                # 左边界收缩
                left = new_left
            if self.is_left_of(apex, right, new_right):
                # 右边界收缩
                right = new_right

            # 如果左右边界交叉，添加新顶点
            if self.is_right_of(apex, left, right):
                # 选择离前一个顶点更近的边界
                if self.distance_squared(apex, left) < self.distance_squared(apex, right):
                    path.append(left)
                    apex = left
                else:
                    path.append(right)
                    apex = right

                # 重置边界
                left = apex
                right = apex

                # 重新开始
                i = path.index(apex)

        # 添加终点
        path.append(portals[-1][0])

        return path
```

### 3.3 局部避障

```python
class LocalAvoidance:
    def __init__(self, max_speed=5.0, max_acceleration=10.0):
        self.max_speed = max_speed
        self.max_acceleration = max_acceleration

    def compute_velocity(self, entity, target_velocity, neighbors, obstacles):
        """计算避免碰撞的速度"""
        # 首选速度（朝向目标）
        preferred_velocity = target_velocity.clamp(self.max_speed)

        # 初始候选速度
        candidate_velocity = preferred_velocity

        # 考虑邻居的影响
        for neighbor in neighbors:
            # 计算相对位置和速度
            relative_pos = neighbor.position - entity.position
            relative_vel = neighbor.velocity - entity.velocity

            # 计算避免碰撞所需的速度变化
            avoidance = self.compute_avoidance(relative_pos, relative_vel,
                                              entity.radius + neighbor.radius)

            # 调整候选速度
            candidate_velocity += avoidance

        # 考虑障碍物的影响
        for obstacle in obstacles:
            # 计算到障碍物的最近点
            closest_point = obstacle.closest_point(entity.position)

            # 计算避免障碍物所需的速度变化
            avoidance = self.compute_obstacle_avoidance(entity.position, closest_point,
                                                       entity.radius)

            # 调整候选速度
            candidate_velocity += avoidance

        # 限制速度和加速度
        candidate_velocity = candidate_velocity.clamp(self.max_speed)

        # 限制加速度
        max_delta = self.max_acceleration * delta_time
        delta = candidate_velocity - entity.velocity
        if delta.length() > max_delta:
            delta = delta.normalize() * max_delta

        return entity.velocity + delta

    def compute_avoidance(self, relative_pos, relative_vel, combined_radius):
        """计算避免其他实体的速度变化"""
        distance = relative_pos.length()
        time_to_collision = 0

        if relative_vel.length() > 0:
            # 计算到最近点的时间
            t = -relative_pos.dot(relative_vel) / relative_vel.length_squared()

            if 0 <= t <= 1.0:  # 在接下来的一帧内可能碰撞
                # 计算最近点
                closest_approach = relative_pos + relative_vel * t
                distance_at_closest = closest_approach.length()

                if distance_at_closest < combined_radius:
                    # 计算碰撞时间
                    a = relative_vel.length_squared()
                    b = 2 * relative_pos.dot(relative_vel)
                    c = relative_pos.length_squared() - combined_radius * combined_radius

                    discriminant = b * b - 4 * a * c
                    if discriminant >= 0:
                        t1 = (-b - math.sqrt(discriminant)) / (2 * a)
                        t2 = (-b + math.sqrt(discriminant)) / (2 * a)
                        time_to_collision = max(t1, t2)

        # 计算避免力
        if time_to_collision > 0:
            # 需要避免的方向
            avoidance_dir = relative_pos.normalize()

            # 避免力的强度（越近越强，越快碰撞越强）
            strength = min(1.0, combined_radius / distance) * (1.0 / (time_to_collision + 0.1))

            return avoidance_dir * strength * self.max_acceleration

        return Vector2(0, 0)
```

### 4.1 目标导向行为（GOAP）

```python
class GOAPAction:
    def __init__(self, name, cost=1.0):
        self.name = name
        self.cost = cost
        self.preconditions = {}  # 执行前需要满足的条件
        self.effects = {}       # 执行后的效果

    def check_preconditions(self, world_state):
        """检查前置条件是否满足"""
        for key, value in self.preconditions.items():
            if key not in world_state or world_state[key] != value:
                return False
        return True

    def apply_effects(self, world_state):
        """应用效果到世界状态"""
        new_state = world_state.copy()
        for key, value in self.effects.items():
            new_state[key] = value
        return new_state

    def execute(self, entity):
        """执行动作"""
        raise NotImplementedError

class GOAPPlanner:
    def __init__(self):
        self.actions = []

    def plan(self, start_state, goal_state, available_actions):
        """规划从起始状态到目标状态的行动序列"""
        # 使用A*搜索行动序列
        open_set = []
        heapq.heappush(open_set, (0, start_state, [], 0))  # (f, state, path, g)

        closed_set = set()

        while open_set:
            f, current_state, path, g = heapq.heappop(open_set)

            # 检查是否达到目标
            if self.state_satisfies_goal(current_state, goal_state):
                return path

            # 添加到关闭列表
            state_key = tuple(sorted(current_state.items()))
            if state_key in closed_set:
                continue
            closed_set.add(state_key)

            # 尝试所有可用动作
            for action in available_actions:
                # 检查前置条件
                if not action.check_preconditions(current_state):
                    continue

                # 应用效果，得到新状态
                new_state = action.apply_effects(current_state)

                # 计算成本
                new_g = g + action.cost
                new_h = self.heuristic(new_state, goal_state)
                new_f = new_g + new_h

                # 添加到开放列表
                new_path = path + [action]
                heapq.heappush(open_set, (new_f, new_state, new_path, new_g))

        return None  # 未找到计划

    def state_satisfies_goal(self, state, goal):
        """检查状态是否满足目标"""
        for key, value in goal.items():
            if key not in state or state[key] != value:
                return False
        return True

    def heuristic(self, state, goal):
        """启发式函数：估计从当前状态到目标的距离"""
        # 简单实现：统计不满足的目标条件数量
        distance = 0
        for key, value in goal.items():
            if key not in state or state[key] != value:
                distance += 1
        return distance
```

### 4.2 实用理论（Utility Theory）

```python
class UtilitySystem:
    def __init__(self):
        self.considerations = []
        self.actions = []

    class Consideration:
        def __init__(self, name, curve_func, weight=1.0):
            self.name = name
            self.curve_func = curve_func  # 输入[0,1]，输出[0,1]
            self.weight = weight

        def evaluate(self, entity):
            """评估考虑因素，返回0-1的值"""
            raise NotImplementedError

        def get_score(self, entity):
            raw_value = self.evaluate(entity)
            curved_value = self.curve_func(raw_value)
            return curved_value * self.weight

    class Action:
        def __init__(self, name):
            self.name = name
            self.considerations = []

        def add_consideration(self, consideration):
            self.considerations.append(consideration)

        def get_score(self, entity):
            """计算动作的总分"""
            if not self.considerations:
                return 0.0

            # 乘法组合（所有考虑因素都影响最终分数）
            product = 1.0
            for consideration in self.considerations:
                score = consideration.get_score(entity)
                product *= score

            # 归一化
            normalization = 1.0 / len(self.considerations)
            final_score = product ** normalization

            return final_score

        def execute(self, entity):
            """执行动作"""
            raise NotImplementedError

    def select_action(self, entity):
        """选择最高分数的动作"""
        best_action = None
        best_score = -1.0

        for action in self.actions:
            score = action.get_score(entity)
            if score > best_score:
                best_score = score
                best_action = action

        return best_action, best_score

# 示例：战士AI的实用系统
def build_warrior_utility_system():
    system = UtilitySystem()

    # 创建考虑因素
    health_consideration = UtilitySystem.Consideration(
        "Health",
        lambda x: 1.0 - x,  # 生命值越低，分数越高（越需要治疗）
        weight=0.8
    )

    enemy_distance_consideration = UtilitySystem.Consideration(
        "EnemyDistance",
        lambda x: 1.0 - x,  # 敌人越近，分数越高
        weight=0.6
    )

    has_ammo_consideration = UtilitySystem.Consideration(
        "HasAmmo",
        lambda x: 1.0 if x > 0 else 0.0,  # 有弹药为1，无弹药为0
        weight=0.4
    )

    # 创建动作
    attack_action = UtilitySystem.Action("Attack")
    attack_action.add_consideration(enemy_distance_consideration)
    attack_action.add_consideration(has_ammo_consideration)

    heal_action = UtilitySystem.Action("Heal")
    heal_action.add_consideration(health_consideration)

    flee_action = UtilitySystem.Action("Flee")
    flee_action.add_consideration(health_consideration)
    flee_action.add_consideration(enemy_distance_consideration)

    # 添加到系统
    system.actions = [attack_action, heal_action, flee_action]

    return system
```

### 5.1 Boids算法

```python
class Boid:
    def __init__(self, position, velocity):
        self.position = position
        self.velocity = velocity
        self.acceleration = Vector2(0, 0)

        # Boids参数
        self.max_speed = 5.0
        self.max_force = 0.2
        self.perception_radius = 50.0

        # 规则权重
        self.separation_weight = 1.5
        self.alignment_weight = 1.0
        self.cohesion_weight = 1.0

    def update(self, delta_time, all_boids):
        # 应用Boids规则
        separation = self.separation(all_boids)
        alignment = self.alignment(all_boids)
        cohesion = self.cohesion(all_boids)

        # 加权求和
        self.acceleration = Vector2(0, 0)
        self.acceleration += separation * self.separation_weight
        self.acceleration += alignment * self.alignment_weight
        self.acceleration += cohesion * self.cohesion_weight

        # 更新速度和位置
        self.velocity += self.acceleration * delta_time
        self.velocity = self.velocity.clamp(self.max_speed)
        self.position += self.velocity * delta_time

    def separation(self, boids):
        """分离规则：避免与邻居太近"""
        steering = Vector2(0, 0)
        total = 0

        for other in boids:
            distance = self.position.distance_to(other.position)
            if other != self and distance < self.perception_radius:
                # 计算远离的方向
                diff = self.position - other.position
                diff = diff.normalize() / distance  # 越近，影响越大
                steering += diff
                total += 1

        if total > 0:
            steering /= total
            if steering.length() > 0:
                steering = steering.normalize() * self.max_speed - self.velocity
                steering = steering.clamp(self.max_force)

        return steering

    def alignment(self, boids):
        """对齐规则：与邻居平均速度对齐"""
        steering = Vector2(0, 0)
        total = 0

        for other in boids:
            distance = self.position.distance_to(other.position)
            if other != self and distance < self.perception_radius:
                steering += other.velocity
                total += 1

        if total > 0:
            steering /= total
            steering = steering.normalize() * self.max_speed - self.velocity
            steering = steering.clamp(self.max_force)

        return steering

    def cohesion(self, boids):
        """凝聚规则：向邻居平均位置移动"""
        steering = Vector2(0, 0)
        total = 0

        for other in boids:
            distance = self.position.distance_to(other.position)
            if other != self and distance < self.perception_radius:
                steering += other.position
                total += 1

        if total > 0:
            steering /= total
            steering = steering - self.position
            steering = steering.normalize() * self.max_speed - self.velocity
            steering = steering.clamp(self.max_force)

        return steering
```

```python
class SquadAI:
    def __init__(self, members):
        self.members = members
        self.formation = None
        self.tactic = "advance"
        self.target = None

    def update(self, delta_time):
        # 根据战术选择行为
        if self.tactic == "advance":
            self.advance_tactic()
        elif self.tactic == "flank":
            self.flank_tactic()
        elif self.tactic == "defend":
            self.defend_tactic()
        elif self.tactic == "retreat":
            self.retreat_tactic()

        # 更新成员
        for member in self.members:
            member.update(delta_time)

    def advance_tactic(self):
        """前进战术：保持阵型向目标前进"""
        if not self.target:
            return

        # 计算阵型位置
        formation_positions = self.formation.calculate_positions(
            self.target.position,
            len(self.members)
        )

        # 分配位置给成员
        for i, member in enumerate(self.members):
            target_pos = formation_positions[i % len(formation_positions)]
            member.move_to(target_pos)

            # 如果接近敌人，开火
            if member.distance_to_enemy() < member.attack_range:
                member.attack()

    def flank_tactic(self):
        """侧翼战术：分两翼包围敌人"""
        if not self.target or len(self.members) < 2:
            return

        # 将小队分为左右两翼
        mid = len(self.members) // 2
        left_wing = self.members[:mid]
        right_wing = self.members[mid:]

        # 计算侧翼位置
        enemy_pos = self.target.position
        enemy_forward = self.target.forward

        # 左翼：敌人左侧
        left_flank = enemy_pos + enemy_forward.rotate(90) * 100
        # 右翼：敌人右侧
        right_flank = enemy_pos + enemy_forward.rotate(-90) * 100

        # 移动两翼
        self.move_wing(left_wing, left_flank, enemy_pos)
        self.move_wing(right_wing, right_flank, enemy_pos)

    def move_wing(self, wing, flank_pos, enemy_pos):
        """移动侧翼"""
        # 计算包围阵型
        for i, member in enumerate(wing):
            # 在侧翼位置周围散开
            angle = (i / len(wing)) * 180 - 90  # -90到90度
            offset = Vector2.from_angle(angle) * 30
            target_pos = flank_pos + offset

            member.move_to(target_pos)

            # 如果面对敌人，开火
            to_enemy = enemy_pos - member.position
            if to_enemy.length() < member.attack_range:
                member.attack()

    def change_tactic(self, new_tactic):
        """改变战术"""
        self.tactic = new_tactic

        # 战术转换的平滑过渡
        for member in self.members:
            member.clear_path()  # 清除当前路径
            member.set_behavior(new_tactic)
```

```python
class AIScheduler:
    def __init__(self, update_rates):
        """
        update_rates: 不同LOD级别的更新频率（秒）
        例如: {"high": 0.1, "medium": 0.5, "low": 1.0}
        """
        self.update_rates = update_rates
        self.entities = []
        self.timers = {}

    def add_entity(self, entity, lod_level="medium"):
        """添加AI实体"""
        self.entities.append(entity)
        self.timers[id(entity)] = {
            "lod": lod_level,
            "accumulator": random.random() * self.update_rates[lod_level]  # 随机偏移
        }

    def update(self, delta_time):
        """更新所有AI实体（根据LOD级别）"""
        for entity in self.entities:
            entity_data = self.timers[id(entity)]
            lod_level = entity_data["lod"]
            update_rate = self.update_rates[lod_level]

            # 累积时间
            entity_data["accumulator"] += delta_time

            # 检查是否需要更新
            if entity_data["accumulator"] >= update_rate:
                entity_data["accumulator"] = 0

                # 执行AI更新
                entity.update_ai(delta_time)

                # 动态调整LOD级别
                self.adjust_lod_level(entity, entity_data)

    def adjust_lod_level(self, entity, entity_data):
        """动态调整AI的LOD级别"""
        distance_to_player = entity.distance_to_player()
        importance = entity.get_importance()

        # 根据距离和重要性确定LOD
        if distance_to_player < 20 and importance > 0.7:
            new_lod = "high"
        elif distance_to_player < 50 and importance > 0.3:
            new_lod = "medium"
        else:
            new_lod = "low"

        # 如果LOD改变，重置计时器
        if new_lod != entity_data["lod"]:
            entity_data["lod"] = new_lod
            entity_data["accumulator"] = 0
```

### 6.2 感知系统优化

```python
class OptimizedPerceptionSystem:
    def __init__(self, world):
        self.world = world
        self.spatial_hash = SpatialHash(cell_size=100)

    def update_vision(self, viewer, potential_targets):
        """优化版的视觉检测"""
        results = []

        # 1. 空间分区快速筛选
        nearby_targets = self.spatial_hash.get_nearby(viewer.position, viewer.vision_range)

        for target in nearby_targets:
            # 2. 距离检查（快速拒绝）
            distance = viewer.position.distance_to(target.position)
            if distance > viewer.vision_range:
                continue

            # 3. 视锥检查（比光线投射快）
            if not self.in_view_cone(viewer, target.position):
                continue

            # 4. 光线投射（最昂贵，最后做）
            if not self.line_of_sight(viewer, target):
                continue

            results.append(target)

        return results

    def in_view_cone(self, viewer, target_pos):
        """检查目标是否在视锥内"""
        # 计算方向向量
        to_target = target_pos - viewer.position
        distance = to_target.length()

        if distance == 0:
            return True

        # 归一化
        to_target_normalized = to_target / distance

        # 计算夹角
        dot_product = viewer.forward.dot(to_target_normalized)
        angle = math.acos(dot_product)

        # 检查是否在视野角度内
        return angle <= viewer.vision_angle / 2

    def line_of_sight(self, viewer, target):
        """光线投射检查视线"""
        # 使用DDA算法进行网格光线投射
        start = viewer.position
        end = target.position

        # 转换为网格坐标
        start_cell = self.world_pos_to_cell(start)
        end_cell = self.world_pos_to_cell(end)

        # DDA算法遍历网格
        dx = abs(end_cell.x - start_cell.x)
        dy = abs(end_cell.y - start_cell.y)

        x = start_cell.x
        y = start_cell.y

        n = 1 + dx + dy

        x_inc = 1 if end_cell.x > start_cell.x else -1
        y_inc = 1 if end_cell.y > start_cell.y else -1

        error = dx - dy

        dx *= 2
        dy *= 2

        for _ in range(n):
            # 检查当前单元格是否有障碍
            if self.world.is_cell_blocked(x, y):
                return False

            if error > 0:
                x += x_inc
                error -= dy
            else:
                y += y_inc
                error += dx

        return True
```

```python
class AILODSystem:
    def __init__(self):
        self.levels = {
            "full": {
                "pathfinding": True,
                "perception": True,
                "decision": True,
                "animation": True,
                "update_rate": 0.1
            },
            "medium": {
                "pathfinding": True,
                "perception": True,
                "decision": True,
                "animation": False,
                "update_rate": 0.3
            },
            "low": {
                "pathfinding": False,  # 使用简单移动
                "perception": False,   # 使用触发器
                "decision": False,     # 使用预设行为
                "animation": False,
                "update_rate": 1.0
            },
            "sleep": {
                "pathfinding": False,
                "perception": False,
                "decision": False,
                "animation": False,
                "update_rate": 5.0
            }
        }

    def get_lod_level(self, entity, player_distance, importance):
        """确定AI的LOD级别"""
        if player_distance < 10:
            return "full"
        elif player_distance < 30:
            return "medium"
        elif player_distance < 100:
            return "low"
        else:
            return "sleep"

    def apply_lod(self, entity, lod_level):
        """应用LOD设置到AI实体"""
        settings = self.levels[lod_level]

        # 应用设置
        entity.ai_component.update_rate = settings["update_rate"]
        entity.ai_component.enable_pathfinding = settings["pathfinding"]
        entity.ai_component.enable_perception = settings["perception"]
        entity.ai_component.enable_decision = settings["decision"]
        entity.animation_component.enabled = settings["animation"]

        # 简化行为
        if lod_level in ["low", "sleep"]:
            entity.ai_component.simplify_behavior()
```

## 总结

游戏AI编程是创造智能、可信赖的游戏角色的关键。从简单的状态机到复杂的行为树和规划系统，AI技术为游戏世界注入了生命。

**关键要点**：
1. **选择合适的技术**：根据AI复杂度选择FSM、行为树或规划系统
2. **性能意识**：AI通常是CPU密集型，需要精心优化
3. **调试优先**：AI行为难以调试，需要强大的可视化工具
4. **迭代开发**：AI需要大量调整和优化才能达到预期效果

**最佳实践**：
- 从简单开始，逐步增加复杂度
- 使用现有的AI中间件（如Behavior Designer、RAIN AI）
- 实现详细的AI调试和可视化
- 与游戏设计团队紧密合作，确保AI支持游戏玩法

**进阶方向**：
- 机器学习驱动的AI（如强化学习）
- 情感和个性建模
- 自适应难度调整
- 社交AI和对话系统

记住：好的游戏AI不是要打败玩家，而是要提供有趣、有挑战性的体验。AI应该支持游戏玩法，而不是成为障碍。
