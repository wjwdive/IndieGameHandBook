# 性能优化与调试

## 概述

本章将深入探讨游戏性能优化和调试的核心技术。游戏性能直接影响玩家体验，从帧率稳定到加载时间，每个环节都需要精心优化。我们将从性能分析工具开始，逐步深入到CPU、GPU、内存等关键优化领域，并提供系统的调试方法论，帮助你构建高效、稳定的游戏。

**学习目标**：
- 掌握游戏性能分析工具的使用方法和数据分析技巧
- 理解CPU性能优化的核心原则和技术（如多线程、批处理、算法优化）
- 掌握GPU性能优化的关键策略（如渲染状态优化、着色器优化、渲染管线优化）
- 学习内存管理和资源优化的最佳实践
- 掌握系统化的游戏调试方法和工具链建设
- 了解平台特定的性能考虑和优化技术

```python
class PerformanceMonitor:
    def __init__(self):
        self.frame_times = CircularBuffer(300)  # 保留最近300帧数据
        self.section_times = {}  # 代码段执行时间
        self.memory_usage = {}
        self.gpu_stats = {}

        # 性能计数器
        self.counters = {
            "draw_calls": 0,
            "triangles": 0,
            "vertices": 0,
            "textures_loaded": 0,
            "shader_switches": 0,
            "buffer_binds": 0
        }

    def begin_frame(self):
        """开始一帧的性能记录"""
        self.frame_start = time.perf_counter()

        # 重置计数器
        for key in self.counters:
            self.counters[key] = 0

    def end_frame(self):
        """结束一帧的性能记录"""
        frame_time = time.perf_counter() - self.frame_start
        self.frame_times.add(frame_time)

        # 计算性能指标
        self.calculate_metrics()

    def begin_section(self, section_name):
        """开始代码段计时"""
        self.section_times[section_name] = {
            "start": time.perf_counter(),
            "calls": self.section_times.get(section_name, {}).get("calls", 0) + 1
        }

    def end_section(self, section_name):
        """结束代码段计时"""
        if section_name in self.section_times:
            section = self.section_times[section_name]
            duration = time.perf_counter() - section["start"]

            # 更新统计
            if "total_time" not in section:
                section["total_time"] = 0
                section["min_time"] = float('inf')
                section["max_time"] = 0

            section["total_time"] += duration
            section["min_time"] = min(section["min_time"], duration)
            section["max_time"] = max(section["max_time"], duration)
            section["avg_time"] = section["total_time"] / section["calls"]

    def calculate_metrics(self):
        """计算性能指标"""
        if len(self.frame_times) == 0:
            return

        # 帧率相关
        self.metrics = {
            "fps": 1.0 / self.frame_times.average() if self.frame_times.average() > 0 else 0,
            "frame_time_avg": self.frame_times.average() * 1000,  # ms
            "frame_time_min": self.frame_times.min() * 1000,
            "frame_time_max": self.frame_times.max() * 1000,
            "frame_time_std": self.frame_times.std() * 1000,

            # 性能百分位（P95, P99）
            "p95_frame_time": self.frame_times.percentile(95) * 1000,
            "p99_frame_time": self.frame_times.percentile(99) * 1000,

            # 卡顿检测（帧时间超过阈值）
            "hitches": len([t for t in self.frame_times.data if t > 0.033])  # >30ms
        }

    def get_performance_report(self):
        """生成性能报告"""
        report = {
            "frame_metrics": self.metrics,
            "section_times": self.section_times,
            "counters": self.counters,
            "memory_usage": self.memory_usage,
            "recommendations": self.generate_recommendations()
        }

        return report

    def generate_recommendations(self):
        """生成优化建议"""
        recommendations = []

        # 基于分析结果生成建议
        if self.metrics["fps"] < 60:
            recommendations.append("帧率低于60FPS，需要优化")

        if self.counters["draw_calls"] > 1000:
            recommendations.append(f"绘制调用过高: {self.counters['draw_calls']}，考虑批次合并")

        if self.section_times.get("physics_update", {}).get("avg_time", 0) > 5:  # >5ms
            recommendations.append("物理更新耗时过长，考虑优化或降低频率")

        return recommendations
```

```python
        """分析捕获文件"""
        if not os.path.exists(capture_file):
            return None

        analysis = {}

        if self.profiler_name == "renderdoc":
            analysis = self.analyze_renderdoc(capture_file)
        elif self.profiler_name == "tracy":
            analysis = self.analyze_tracy(capture_file)

        return analysis

    def analyze_renderdoc(self, rdc_file):
        """分析RenderDoc捕获"""
        # 使用RenderDoc Python API分析
        import renderdoc

        analysis = {
            "draw_calls": 0,
            "api_calls": 0,
            "texture_memory": 0,
            "buffer_memory": 0,
            "pipeline_states": 0,
            "events": []
        }

        # 打开捕获文件
        cap = renderdoc.OpenCaptureFile()
        status = cap.OpenFile(rdc_file, '', None)

        if status == renderdoc.ReplayStatus.Succeeded:
            # 分析每一帧
            for frame in range(cap.GetNumFrames()):
                frame_data = cap.GetFrameInfo(frame)

                analysis["api_calls"] += frame_data.dispatchs

                # 分析资源使用
                for res in frame_data.resources:
                    if res.type == renderdoc.ResourceType.Texture:
                        analysis["texture_memory"] += res.byteSize
                    elif res.type == renderdoc.ResourceType.Buffer:
                        analysis["buffer_memory"] += res.byteSize
```

```python
class PerformanceBenchmark:
    def __init__(self):
        self.scenarios = []
        self.results = {}

    def add_scenario(self, name, setup_func, run_func, teardown_func):
        """添加测试场景"""
        self.scenarios.append({
            "name": name,
            "setup": setup_func,
            "run": run_func,
            "teardown": teardown_func
        })

    def run_benchmark(self, iterations=100, warmup=10):
        """运行基准测试"""
        all_results = {}

        for scenario in self.scenarios:
            print(f"Running benchmark: {scenario['name']}")

            # 预热
            scenario['setup']()
            for _ in range(warmup):
                scenario['run']()

            # 正式测试
            times = []
            for i in range(iterations):
                start = time.perf_counter()
                scenario['run']()
                end = time.perf_counter()
                times.append((end - start) * 1000)  # 转换为ms

            # 清理
            scenario['teardown']()

            # 分析结果
            scenario_results = {
                "iterations": iterations,
                "times": times,
                "avg": np.mean(times),
                "min": np.min(times),
                "max": np.max(times),
                "std": np.std(times),
                "p95": np.percentile(times, 95),
                "p99": np.percentile(times, 99)
            }

            all_results[scenario['name']] = scenario_results

        self.results = all_results
        return all_results

    def compare_scenarios(self):
        """比较不同场景的性能"""
        if not self.results:
            return None

        comparison = {}

        # 找出基准场景（通常是第一个）
        baseline_name = list(self.results.keys())[0]
        baseline = self.results[baseline_name]

        for name, results in self.results.items():
            if name == baseline_name:
                continue

            # 计算相对于基准的性能差异
            speedup = baseline["avg"] / results["avg"] if results["avg"] > 0 else 0
            improvement = (baseline["avg"] - results["avg"]) / baseline["avg"] * 100

            comparison[name] = {
                "speedup": speedup,
                "improvement_percent": improvement,
                "is_better": results["avg"] < baseline["avg"]
            }

        return comparison

    def generate_report(self, output_file="benchmark_report.md"):
        """生成基准测试报告"""
        report = "# 性能基准测试报告\n\n"
        report += f"生成时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n"

        # 总体结果
        report += "## 总体结果\n\n"
        for name, results in self.results.items():
            report += f"### {name}\n"
            report += f"- 平均时间: {results['avg']:.2f}ms\n"
            report += f"- 最小时间: {results['min']:.2f}ms\n"
            report += f"- 最大时间: {results['max']:.2f}ms\n"
            report += f"- 标准差: {results['std']:.2f}ms\n"
            report += f"- P95: {results['p95']:.2f}ms\n"
            report += f"- P99: {results['p99']:.2f}ms\n\n"

        # 比较结果
        comparison = self.compare_scenarios()
        if comparison:
            report += "## 性能比较\n\n"
            for name, comp in comparison.items():
                better_text = "更好" if comp["is_better"] else "更差"
                report += f"- {name}: {better_text}，加速比: {comp['speedup']:.2f}x，改进: {comp['improvement_percent']:.1f}%\n"

        # 建议
        report += "\n## 优化建议\n\n"
        report += self.generate_optimization_suggestions()

        # 保存报告
        with open(output_file, 'w') as f:
            f.write(report)

        return report
```

```python
class JobSystem:
    def __init__(self, num_worker_threads=None):
        # 根据CPU核心数确定工作线程数
        if num_worker_threads is None:
            num_worker_threads = max(1, os.cpu_count() - 1)  # 保留一个核心给主线程

        self.num_workers = num_worker_threads
        self.worker_threads = []
        self.job_queue = queue.Queue()
        self.job_id_counter = 0
        self.completed_jobs = {}
        self.running = False

        # 统计信息
        self.stats = {
            "jobs_submitted": 0,
            "jobs_completed": 0,
            "jobs_failed": 0,
            "average_job_time": 0,
            "worker_utilization": [0] * num_worker_threads
        }

    def start(self):
        """启动工作线程"""
        self.running = True

        for i in range(self.num_workers):
            thread = threading.Thread(target=self.worker_loop, args=(i,))
            thread.daemon = True
            thread.start()
            self.worker_threads.append(thread)

    def stop(self):
        """停止工作线程"""
        self.running = False

        # 添加停止信号
        for _ in range(self.num_workers):
            self.job_queue.put(None)

        # 等待线程结束
        for thread in self.worker_threads:
            thread.join()

    def worker_loop(self, worker_id):
        """工作线程主循环"""
        while self.running:
            try:
                job = self.job_queue.get(timeout=0.1)

                if job is None:  # 停止信号
                    break

                # 执行作业
                start_time = time.perf_counter()
                try:
                    result = job["func"](*job["args"], **job["kwargs"])
                    self.completed_jobs[job["id"]] = {
                        "result": result,
                        "worker_id": worker_id,
                        "execution_time": time.perf_counter() - start_time
                    }

                    self.stats["jobs_completed"] += 1

                except Exception as e:
                    self.completed_jobs[job["id"]] = {
                        "error": str(e),
                        "worker_id": worker_id,
                        "execution_time": time.perf_counter() - start_time
                    }
                    self.stats["jobs_failed"] += 1

                finally:
                    self.job_queue.task_done()

                # 更新统计
                self.update_worker_stats(worker_id, start_time)

            except queue.Empty:
                continue

    def submit_job(self, func, *args, **kwargs):
        """提交作业到队列"""
        job_id = self.job_id_counter
        self.job_id_counter += 1

        job = {
            "id": job_id,
            "func": func,
            "args": args,
            "kwargs": kwargs,
            "submit_time": time.perf_counter()
        }

        self.job_queue.put(job)
        self.stats["jobs_submitted"] += 1

        return job_id

    def wait_for_job(self, job_id, timeout=None):
        """等待作业完成"""
        start_time = time.perf_counter()

        while job_id not in self.completed_jobs:
            if timeout and (time.perf_counter() - start_time) > timeout:
                raise TimeoutError(f"Job {job_id} timeout")
            time.sleep(0.001)  # 短暂休眠避免忙等待

        return self.completed_jobs[job_id]

    def parallel_for(self, items, process_func, chunk_size=100):
        """并行处理列表中的元素"""
        num_chunks = max(1, len(items) // chunk_size)
        chunks = [items[i:i+chunk_size] for i in range(0, len(items), chunk_size)]

        job_ids = []
        for chunk in chunks:
            job_id = self.submit_job(self._process_chunk, process_func, chunk)
            job_ids.append(job_id)

        # 等待所有作业完成
        results = []
        for job_id in job_ids:
            result = self.wait_for_job(job_id)
            if "result" in result:
                results.extend(result["result"])

        return results

    def _process_chunk(self, process_func, chunk):
        """处理数据块（在工作线程中执行）"""
        return [process_func(item) for item in chunk]
```

```python
class CacheOptimizer:

        # 分析访问模式
        access_analysis = self.analyze_access_patterns(access_patterns)

        # 重新组织数据
        optimized_arrays = {}

        for array_name, data in data_arrays.items():
            if array_name in access_analysis["hot"]:
                # 热数据：确保连续存储
                optimized = self.make_contiguous(data)
            elif array_name in access_analysis["cold"]:
                # 冷数据：可以分块存储
                optimized = self.chunk_data(data, self.cache_line_size)
            else:
                # 一般数据：保持原样或简单优化
                optimized = data

            optimized_arrays[array_name] = optimized

        return optimized_arrays

    def analyze_access_patterns(self, access_patterns):
        """分析数据访问模式"""
        analysis = {
            "hot": set(),  # 频繁访问的数据
            "cold": set(),  # 很少访问的数据
            "sequential": set(),  # 顺序访问的数据
            "random": set()  # 随机访问的数据
        }

        for pattern in access_patterns:
            data_name = pattern["data"]
            frequency = pattern.get("frequency", 1)
            access_type = pattern.get("access_type", "random")

            if frequency > 100:  # 高频访问
                analysis["hot"].add(data_name)
            elif frequency < 10:  # 低频访问
                analysis["cold"].add(data_name)

            analysis[access_type].add(data_name)

        return analysis

    def make_contiguous(self, data):
        """确保数据连续存储"""
        if isinstance(data, list):
            # 列表已经是连续的
            return data
        elif hasattr(data, 'flags'):
            # NumPy数组：确保C连续
            return np.ascontiguousarray(data)
        else:
            # 其他类型：转换为列表
            return list(data)

    def chunk_data(self, data, chunk_size):
        """将数据分块存储"""
        if len(data) <= chunk_size:
            return data

        chunks = []
        for i in range(0, len(data), chunk_size):
            chunk = data[i:i+chunk_size]

            # 确保每个块是缓存行对齐的
            aligned_chunk = self.align_to_cache_line(chunk)
            chunks.append(aligned_chunk)

        return chunks

    def align_to_cache_line(self, data):
        """数据缓存行对齐"""
        # 计算需要填充的字节数
        data_size = len(data) if hasattr(data, '__len__') else sys.getsizeof(data)
        padding = (self.cache_line_size - (data_size % self.cache_line_size)) % self.cache_line_size

        if padding > 0:
            # 添加填充
            if isinstance(data, bytearray):
                data.extend(b'\x00' * padding)
            elif isinstance(data, list):
                data.extend([0] * (padding // 4))  # 假设每个元素4字节

        return data

    def prefetch_data(self, data, next_indices):
        """预取可能访问的数据"""
```

```python
class AlgorithmOptimizer:
    def __init__(self):
        self.optimization_patterns = {
            "loop_unrolling": self.optimize_loop_unrolling,
            "branch_prediction": self.optimize_branch_prediction,
            "memory_access": self.optimize_memory_access,
            "algorithm_selection": self.optimize_algorithm_selection
        }

    def optimize_code(self, code_analysis):
        """根据分析结果优化代码"""
        optimizations = []

        # 检查每个优化模式
        for pattern_name, optimizer in self.optimization_patterns.items():
            if self.should_apply_pattern(code_analysis, pattern_name):
                optimized_code = optimizer(code_analysis)
                if optimized_code:
                    optimizations.append({
                        "pattern": pattern_name,
                        "code": optimized_code,
                        "estimated_improvement": self.estimate_improvement(code_analysis, pattern_name)
                    })

        return optimizations

    def optimize_loop_unrolling(self, code_analysis):
        """循环展开优化"""
        if not code_analysis.get("has_loops", False):
            return None

        loops = code_analysis.get("loops", [])
        optimized_loops = []

        for loop in loops:
            if self.is_suitable_for_unrolling(loop):
                unrolled = self.unroll_loop(loop)
                optimized_loops.append(unrolled)

        return optimized_loops if optimized_loops else None

    def is_suitable_for_unrolling(self, loop):
        """判断循环是否适合展开"""
        # 小循环（迭代次数少）
        if loop.get("iterations", 1000) < 100:
            return True

        # 循环体简单（操作少）
        if loop.get("operations_per_iteration", 10) < 5:
            return True

        # 没有复杂控制流
        if not loop.get("has_complex_control_flow", False):
            return True

        return False

    def unroll_loop(self, loop, factor=4):
        """展开循环"""
        original_code = loop["code"]
        # 简化的循环展开示例
        # 实际实现需要解析AST并重写循环

        unrolled_code = f"""
        // 循环展开 (因子={factor})
        for (int i = 0; i < {loop['iterations']}; i += {factor}) {{
            {original_code.replace('i', 'i')}
            if (i + 1 < {loop['iterations']}) {{
                {original_code.replace('i', 'i + 1')}
            }}
            if (i + 2 < {loop['iterations']}) {{
                {original_code.replace('i', 'i + 2')}
            }}
            if (i + 3 < {loop['iterations']}) {{
                {original_code.replace('i', 'i + 3')}
            }}
        }}
        """

        return unrolled_code

    def optimize_branch_prediction(self, code_analysis):
        """分支预测优化"""
        if not code_analysis.get("has_branches", False):
            return None

        branches = code_analysis.get("branches", [])
        optimized_branches = []

        for branch in branches:
            if self.is_predictable_branch(branch):
                optimized = self.reorder_branch(branch)
                optimized_branches.append(optimized)

        return optimized_branches if optimized_branches else None

    def is_predictable_branch(self, branch):
        """判断分支是否可预测"""
        # 分支概率高度偏斜（>90%或<10%）
        probability = branch.get("probability", 0.5)
        return probability > 0.9 or probability < 0.1

    def reorder_branch(self, branch):
        """重新排列分支条件"""
        # 将更可能执行的分支放在前面
        if branch.get("probability", 0.5) > 0.5:
            # 大概率分支在前
            optimized = f"""
            if (likely({branch['condition']})) {{
                {branch['true_block']}
            }} else {{
                {branch['false_block']}
            }}
            """
        else:
            # 小概率分支在前（或使用likely/unlikely宏）
            optimized = f"""
            if (unlikely({branch['condition']})) {{
                {branch['true_block']}
            }} else {{
                {branch['false_block']}
            }}
```

```python
            # 如果状态相同，不需要改变
            if current_value == new_value:

            if key == "shader_program":
                # 着色器程序切换昂贵
                if current_value and new_value and current_value.id == new_value.id:
                    continue

            elif key == "texture":
                # 纹理绑定优化
                if self.can_skip_texture_bind(current_value, new_value):

            # 需要改变状态

        if current_texture.id == new_texture.id:
            return True

        # 如果纹理在同一个纹理数组中（现代API可以快速切换）
        if (hasattr(current_texture, 'array_index') and
            hasattr(new_texture, 'array_index') and
            current_texture.array_id == new_texture.array_id):
            # 只需要改变数组索引，不需要完全重新绑定
            return True

                "state": group["state"]
            })

            # 批处理绘制命令
            if self.can_batch_draws(group["draws"]):
                batched_draw = self.batch_draw_commands(group["draws"])
                optimized_commands.append(batched_draw)
            else:

                optimized_commands.extend(group["draws"])

            return False

        # 检查所有绘制是否使用相同的顶点格式
        first_vertex_format = draws[0].get("vertex_format")
        for draw in draws[1:]:
            if draw.get("vertex_format") != first_vertex_format:
                return False

        # 检查是否都是同一类型的图元
        first_primitive = draws[0].get("primitive_type")
        for draw in draws[1:]:
            if draw.get("primitive_type") != first_primitive:
                return False

        return True

            batched_indices.extend(adjusted_indices)

        return {

            "vertices": batched_vertices,
            "indices": batched_indices,
            "primitive_type": draws[0]["primitive_type"],
            "vertex_format": draws[0]["vertex_format"]
```

```python
        ast = self.remove_unused_declarations(ast, used_variables)

        # 移除无法到达的代码
        ast = self.remove_unreachable_code(ast)

        return ast

    def optimize_constant_folding(self, ast):
        """常量折叠"""
        # 查找常量表达式
        constant_expressions = self.find_constant_expressions(ast)

        # 预计算常量表达式
        for expr in constant_expressions:
            value = self.evaluate_expression(expr)
            if value is not None:
                # 用常量值替换表达式
                ast = self.replace_with_constant(ast, expr, value)

        return ast

        # 查找重复的表达式
        common_exprs = self.find_common_expressions(ast)

        # 为公共表达式创建临时变量
        for expr, locations in common_exprs.items():
            if len(locations) > 1:
                # 创建临时变量
                temp_var = self.create_temp_variable()

                # 在第一个出现位置计算并赋值
                first_location = locations[0]
                ast = self.insert_temp_assignment(ast, first_location, expr, temp_var)
```

```python
class ResourceManager:
    def __init__(self, max_memory_mb=512):
        self.max_memory = max_memory_mb * 1024 * 1024  # 转换为字节
        self.current_memory = 0

        self.resources = {}  # resource_id: ResourceInfo
        self.priority_queue = []  # 按优先级排序的资源ID

        # 加载策略
        self.load_strategies = {
            "eager": self.load_eager,
            "lazy": self.load_lazy,
            "streaming": self.load_streaming
        }

    def load_resource(self, resource_id, path, strategy="lazy", priority=0):
        """加载资源"""
        # 检查是否已加载
        if resource_id in self.resources:
            # 更新访问时间和优先级
            self.update_resource_priority(resource_id, priority)
            return self.resources[resource_id]["data"]

        # 创建资源记录
        resource_info = {
            "id": resource_id,
            "path": path,
            "strategy": strategy,
            "priority": priority,
            "last_access": time.time(),
            "size": 0,
            "data": None,
            "state": "pending"
        }

        self.resources[resource_id] = resource_info

        # 根据策略加载
        loader = self.load_strategies.get(strategy, self.load_lazy)
        loader(resource_info)

        # 添加到优先级队列
        self.priority_queue.append(resource_id)
        self.sort_priority_queue()

        return resource_info["data"]

    def load_eager(self, resource_info):
        """立即加载资源"""
        try:
            # 加载资源数据
            data = self.load_from_disk(resource_info["path"])
            size = len(data) if hasattr(data, '__len__') else sys.getsizeof(data)

            # 检查内存限制
            if not self.check_memory_limit(size):
                # 内存不足，需要卸载一些资源
                self.free_memory(size)

            # 存储资源
            resource_info["data"] = data
            resource_info["size"] = size
            resource_info["state"] = "loaded"

            # 更新内存使用
            self.current_memory += size

        except Exception as e:
            resource_info["state"] = f"error: {str(e)}"

    def load_lazy(self, resource_info):
        """延迟加载（按需加载）"""
        # 只记录元数据，不立即加载数据
        resource_info["state"] = "unloaded"

    def load_streaming(self, resource_info):
        """流式加载"""
        # 启动后台加载
        thread = threading.Thread(target=self.stream_load, args=(resource_info,))
        thread.daemon = True
        thread.start()

    def stream_load(self, resource_info):
        """流式加载实现"""
        resource_info["state"] = "streaming"

        try:
            # 分块加载
            chunk_size = 1024 * 1024  # 1MB chunks
            total_size = self.get_file_size(resource_info["path"])

            resource_info["data"] = bytearray(total_size)
            resource_info["size"] = total_size

            with open(resource_info["path"], 'rb') as f:
                offset = 0
                while offset < total_size:
                    # 读取一块数据
                    chunk = f.read(chunk_size)
                    if not chunk:
                        break

                    # 复制到目标位置
                    resource_info["data"][offset:offset+len(chunk)] = chunk
                    offset += len(chunk)

                    # 更新状态
                    resource_info["stream_progress"] = offset / total_size

                    # 检查内存限制
                    if not self.check_memory_limit(offset):
                        # 暂停加载，等待内存释放
                        time.sleep(0.1)

            resource_info["state"] = "loaded"

        except Exception as e:
            resource_info["state"] = f"error: {str(e)}"

    def unload_resource(self, resource_id):
        """卸载资源"""
        if resource_id not in self.resources:
            return False

        resource_info = self.resources[resource_id]

        if resource_info["state"] == "loaded":
            # 释放内存
            self.current_memory -= resource_info["size"]
            resource_info["data"] = None
            resource_info["state"] = "unloaded"

            # 从优先级队列移除
            if resource_id in self.priority_queue:
                self.priority_queue.remove(resource_id)

            return True

        return False

    def free_memory(self, required_size):
        """释放内存以容纳新资源"""
        # 按优先级排序（低优先级先释放）
        self.sort_priority_queue()

        freed_memory = 0

        # 从低优先级开始释放资源
        for resource_id in reversed(self.priority_queue):
            if freed_memory >= required_size:
                break

            resource_info = self.resources[resource_id]

            if resource_info["state"] == "loaded":
                # 卸载资源
                self.unload_resource(resource_id)
                freed_memory += resource_info["size"]

        return freed_memory

    def check_memory_limit(self, additional_size):
        """检查内存限制"""
        return (self.current_memory + additional_size) <= self.max_memory

    def update_resource_priority(self, resource_id, new_priority=None):
        """更新资源优先级"""
        if resource_id not in self.resources:
            return

        resource_info = self.resources[resource_id]

        if new_priority is not None:
            resource_info["priority"] = new_priority

        # 更新访问时间
        resource_info["last_access"] = time.time()

        # 重新排序优先级队列
        self.sort_priority_queue()

    def sort_priority_queue(self):
        """按优先级排序资源队列"""
        self.priority_queue.sort(key=lambda rid: (
            -self.resources[rid]["priority"],  # 优先级降序
            self.resources[rid]["last_access"]  # 访问时间升序（旧资源先处理）
        ))

    def get_memory_report(self):
        """获取内存使用报告"""
        report = {
            "total_memory": self.max_memory,
            "used_memory": self.current_memory,
            "available_memory": self.max_memory - self.current_memory,
            "usage_percentage": self.current_memory / self.max_memory * 100,
            "loaded_resources": 0,
            "unloaded_resources": 0,
            "streaming_resources": 0,
            "largest_resource": {"id": None, "size": 0}
        }

        for rid, info in self.resources.items():
            if info["state"] == "loaded":
                report["loaded_resources"] += 1

                # 跟踪最大资源
                if info["size"] > report["largest_resource"]["size"]:
                    report["largest_resource"] = {"id": rid, "size": info["size"]}

            elif info["state"] == "unloaded":
                report["unloaded_resources"] += 1
            elif info["state"] == "streaming":
                report["streaming_resources"] += 1

        return report
```

```python
class HotReloadSystem:
    def __init__(self, watch_directories=None):
        if watch_directories is None:
            watch_directories = ["scripts", "shaders", "config"]

        self.watch_directories = watch_directories
        self.watchers = {}
        self.modified_files = {}
        self.reload_handlers = {}

        # 文件修改时间缓存
        self.file_mtimes = {}

    def start_watching(self):
        """开始监视文件变化"""
        for directory in self.watch_directories:
            if not os.path.exists(directory):
                continue

            # 创建文件系统观察器
            watcher = watchfiles.watch(directory)
            self.watchers[directory] = watcher

            # 记录初始修改时间
            self.scan_directory(directory)

        print(f"开始监视 {len(self.watchers)} 个目录的热重载")

    def stop_watching(self):
        """停止监视文件变化"""
        for watcher in self.watchers.values():
            watcher.stop()

        self.watchers.clear()
        print("停止热重载监视")

    def scan_directory(self, directory):
        """扫描目录并记录文件修改时间"""
        for root, dirs, files in os.walk(directory):
            for file in files:
                filepath = os.path.join(root, file)
                try:
                    mtime = os.path.getmtime(filepath)
                    self.file_mtimes[filepath] = mtime
                except OSError:
                    continue

    def check_for_changes(self):
        """检查文件变化"""
        changes_detected = False

        for directory, watcher in self.watchers.items():
            try:
                changes = next(watcher, None)
                if changes:
                    for change in changes:
                        change_type, filepath = change

                        if change_type == watchfiles.Change.modified:
                            # 检查修改时间是否真的改变了
                            current_mtime = os.path.getmtime(filepath)
                            last_mtime = self.file_mtimes.get(filepath, 0)

                            if current_mtime > last_mtime:
                                self.file_mtimes[filepath] = current_mtime
                                self.modified_files[filepath] = current_mtime
                                changes_detected = True

            except StopIteration:
                # 观察器结束
                pass
            except Exception as e:
                print(f"检查文件变化时出错: {e}")

        return changes_detected

    def process_changes(self):
        """处理检测到的文件变化"""
        if not self.modified_files:
            return

        print(f"检测到 {len(self.modified_files)} 个文件变化")

        for filepath, mtime in list(self.modified_files.items()):
            try:
                self.reload_file(filepath)
                del self.modified_files[filepath]

            except Exception as e:
                print(f"重新加载文件 {filepath} 时出错: {e}")

    def reload_file(self, filepath):
        """重新加载文件"""
        # 确定文件类型
        file_ext = os.path.splitext(filepath)[1].lower()

        # 查找对应的重载处理器
        handler = self.get_reload_handler(file_ext)

        if handler:
            print(f"重新加载: {filepath}")
            handler(filepath)
        else:
            print(f"没有找到 {file_ext} 文件类型的重载处理器")

    def get_reload_handler(self, file_ext):
        """获取文件类型的重载处理器"""
        handler_map = {
            ".py": self.reload_python_script,
            ".lua": self.reload_lua_script,
            ".glsl": self.reload_shader,
            ".json": self.reload_config,
            ".xml": self.reload_config,
            ".yaml": self.reload_config,
            ".yml": self.reload_config
        }

        return handler_map.get(file_ext)

    def reload_python_script(self, filepath):
        """重新加载Python脚本"""
        # 解析模块名
        module_name = self.filepath_to_module(filepath)

        if not module_name:
            print(f"无法确定 {filepath} 的模块名")
            return

        # 重新加载模块
        if module_name in sys.modules:
            import importlib
            importlib.reload(sys.modules[module_name])
            print(f"重新加载模块: {module_name}")
        else:
            print(f"模块 {module_name} 未加载")

    def reload_shader(self, filepath):
        """重新加载着色器"""
        # 查找使用此着色器的材质
        materials = self.find_materials_using_shader(filepath)

        for material in materials:
            # 重新编译着色器
            new_shader = self.compile_shader(filepath)

            if new_shader:
                # 更新材质
                material.shader = new_shader
                print(f"更新材质 {material.name} 的着色器")
            else:
                print(f"编译着色器失败: {filepath}")

    def reload_config(self, filepath):
        """重新加载配置文件"""
        # 解析配置文件
        config_data = self.parse_config_file(filepath)

        if config_data:
            # 更新配置系统
            config_system = self.get_config_system()
            config_system.update_from_file(filepath, config_data)
            print(f"更新配置: {filepath}")

    def filepath_to_module(self, filepath):
        """将文件路径转换为模块名"""
        # 移除扩展名
        filepath = os.path.splitext(filepath)[0]

        # 转换为模块路径
        module_parts = []

        # 查找文件所在的包
        current = filepath
        while current:
            parent = os.path.dirname(current)
            basename = os.path.basename(current)

            # 检查是否有 __init__.py
            init_file = os.path.join(current, "__init__.py")
            if os.path.exists(init_file):
                module_parts.insert(0, basename)

            current = parent

            # 检查是否到达项目根目录
            if self.is_project_root(current):
                break

        if not module_parts:
            return None

        return ".".join(module_parts)

    def register_reload_handler(self, file_ext, handler):
        """注册自定义重载处理器"""
        self.reload_handlers[file_ext] = handler
        print(f"注册 {file_ext} 文件类型的重载处理器")

    def update(self):
        """更新热重载系统（每帧调用）"""
        if self.check_for_changes():
            self.process_changes()
```

```python
class MobileOptimizer:
    def __init__(self):
        self.platform_info = self.detect_platform()
        self.optimization_settings = self.get_default_settings()

    def detect_platform(self):
        """检测移动平台"""
        platform_info = {
            "is_mobile": False,
            "platform": "unknown",
            "gpu_tier": 0,
            "memory_mb": 0,
            "cpu_cores": 0
        }

        # 简化检测（实际需要平台特定API）
        try:
            # 假设在Android上
            import android
            platform_info.update({
                "is_mobile": True,
                "platform": "android",
                "memory_mb": android.get_memory_info()["total"] // (1024 * 1024),
                "cpu_cores": android.get_cpu_info()["cores"]
            })
        except ImportError:
            try:
                # 假设在iOS上
                import ios
                platform_info.update({
                    "is_mobile": True,
                    "platform": "ios",
                    "memory_mb": ios.get_memory() // (1024 * 1024),
                    "cpu_cores": ios.get_cpu_cores()
                })
            except ImportError:
                # 桌面平台
                import psutil
                platform_info.update({
                    "is_mobile": False,
                    "platform": "desktop",
                    "memory_mb": psutil.virtual_memory().total // (1024 * 1024),
                    "cpu_cores": os.cpu_count()
                })

        return platform_info

    def get_default_settings(self):
        """获取默认优化设置"""
        settings = {
            "texture_quality": "medium",
            "shadow_quality": "low",
            "particle_quality": "medium",
            "lod_distance": 50.0,
            "max_lights": 4,
            "post_processing": False,
            "anti_aliasing": "none",
            "resolution_scale": 1.0,
            "frame_rate_target": 30,
            "physics_updates": 30
        }

        # 根据平台调整
        if self.platform_info["is_mobile"]:
            memory_mb = self.platform_info["memory_mb"]

            if memory_mb < 1000:  # < 1GB
                settings.update({
                    "texture_quality": "low",
                    "shadow_quality": "off",
                    "particle_quality": "low",
                    "lod_distance": 30.0,
                    "max_lights": 2,
                    "resolution_scale": 0.75,
                    "frame_rate_target": 30
                })
            elif memory_mb < 2000:  # < 2GB
                settings.update({
                    "texture_quality": "medium",
                    "shadow_quality": "low",
                    "particle_quality": "medium",
                    "lod_distance": 50.0,
                    "max_lights": 3,
                    "resolution_scale": 0.85,
                    "frame_rate_target": 45
                })
            else:  # >= 2GB
                settings.update({
                    "texture_quality": "high",
                    "shadow_quality": "medium",
                    "particle_quality": "high",
                    "lod_distance": 70.0,
                    "max_lights": 4,
                    "resolution_scale": 1.0,
                    "frame_rate_target": 60
                })

        return settings

    def apply_optimizations(self, game_systems):
        """应用移动平台优化"""
        optimizations = []

        # 渲染优化
        render_optimizations = self.optimize_rendering(game_systems.get("renderer"))
        optimizations.extend(render_optimizations)

        # 内存优化
        memory_optimizations = self.optimize_memory(game_systems.get("resource_manager"))
        optimizations.extend(memory_optimizations)

        # 性能优化
        performance_optimizations = self.optimize_performance(game_systems)
        optimizations.extend(performance_optimizations)

        # 应用设置
        self.apply_settings(game_systems)

        return optimizations

    def optimize_rendering(self, renderer):
        """优化渲染"""
        optimizations = []

        if not renderer:
            return optimizations

        # 1. 降低纹理质量
        texture_quality = self.optimization_settings["texture_quality"]
        renderer.set_texture_quality(texture_quality)
        optimizations.append(f"纹理质量设置为: {texture_quality}")

        # 2. 减少绘制距离
        lod_distance = self.optimization_settings["lod_distance"]
        renderer.set_lod_distance(lod_distance)
        optimizations.append(f"LOD距离设置为: {lod_distance}")

        # 3. 限制光源数量
        max_lights = self.optimization_settings["max_lights"]
        renderer.set_max_lights(max_lights)
        optimizations.append(f"最大光源数: {max_lights}")

        # 4. 禁用昂贵的后期处理
        if not self.optimization_settings["post_processing"]:
            renderer.disable_post_processing()
            optimizations.append("禁用后期处理")

        # 5. 动态分辨率缩放
        resolution_scale = self.optimization_settings["resolution_scale"]
        if resolution_scale < 1.0:
            renderer.set_resolution_scale(resolution_scale)
            optimizations.append(f"分辨率缩放: {resolution_scale}")

        return optimizations

    def optimize_memory(self, resource_manager):
        """优化内存使用"""
        optimizations = []

        if not resource_manager:
            return optimizations

        # 1. 调整纹理压缩
        compression_level = "high" if self.platform_info["memory_mb"] < 2000 else "medium"
        resource_manager.set_texture_compression(compression_level)
        optimizations.append(f"纹理压缩级别: {compression_level}")

        # 2. 降低模型细节
        if self.platform_info["memory_mb"] < 1000:
            resource_manager.set_max_model_triangles(5000)
            optimizations.append("最大模型三角形数: 5000")
        elif self.platform_info["memory_mb"] < 2000:
            resource_manager.set_max_model_triangles(10000)
            optimizations.append("最大模型三角形数: 10000")

        # 3. 更积极的流式加载
        resource_manager.set_streaming_aggressiveness("high")
        optimizations.append("流式加载积极性: 高")

        # 4. 更小的音频缓冲区
        resource_manager.set_audio_buffer_size(2048)  # 2KB缓冲区
        optimizations.append("音频缓冲区大小: 2KB")

        return optimizations

    def optimize_performance(self, game_systems):
        """优化性能"""
        optimizations = []

        # 1. 调整物理更新频率
        physics_updates = self.optimization_settings["physics_updates"]
        if "physics" in game_systems:
            game_systems["physics"].set_update_frequency(physics_updates)
            optimizations.append(f"物理更新频率: {physics_updates}Hz")

        # 2. 降低AI更新频率
        if "ai" in game_systems:
            game_systems["ai"].set_update_frequency(10)  # 10Hz
            optimizations.append("AI更新频率: 10Hz")

        # 3. 帧率限制
        target_fps = self.optimization_settings["frame_rate_target"]
        if "game_loop" in game_systems:
            game_systems["game_loop"].set_target_fps(target_fps)
            optimizations.append(f"目标帧率: {target_fps}FPS")

        # 4. 启用节能模式
        if "power" in game_systems:
            game_systems["power"].enable_power_saving()
            optimizations.append("启用节能模式")

        return optimizations

    def apply_settings(self, game_systems):
        """应用所有优化设置"""
        # 保存设置供UI显示
        self.save_settings()

        # 通知所有系统
        for system_name, system in game_systems.items():
            if hasattr(system, 'apply_platform_settings'):
                system.apply_platform_settings(self.optimization_settings)

    def monitor_performance(self):
        """监控性能并动态调整"""
        # 获取当前性能指标
        fps = self.get_current_fps()
        memory_usage = self.get_memory_usage()
        thermal_state = self.get_thermal_state()

        # 动态调整设置
        adjustments = []

        # 帧率过低
        if fps < self.optimization_settings["frame_rate_target"] * 0.8:
            # 进一步降低质量
            if self.optimization_settings["resolution_scale"] > 0.5:
                self.optimization_settings["resolution_scale"] *= 0.9
                adjustments.append(f"降低分辨率缩放: {self.optimization_settings['resolution_scale']:.2f}")

        # 内存使用过高
        if memory_usage > 0.8:  # >80%
            # 减少纹理质量
            if self.optimization_settings["texture_quality"] != "low":
                self.optimization_settings["texture_quality"] = "low"
                adjustments.append("纹理质量降为: low")

        # 温度过高
        if thermal_state == "critical":
            # 启用严格的节能措施
            self.optimization_settings["frame_rate_target"] = 30
            self.optimization_settings["physics_updates"] = 20
            adjustments.append("温度过高，启用严格节能")

        return adjustments
```

**关键要点**：
1. **测量优先**：优化前先测量，找到真正的性能瓶颈
2. **分层优化**：从算法到微架构，从高层次到低层次逐步优化
3. **平台感知**：不同平台有不同的优化策略和限制
4. **持续监控**：性能优化不是一次性的，需要持续监控和调整

**最佳实践**：
- 建立完整的性能分析工具链
- 实现详细的游戏内调试系统
- 为不同平台预设优化配置
- 进行定期的性能回归测试
- 保持代码的可优化性（避免过度抽象）

**进阶方向**：
- 机器学习驱动的性能优化
- 实时自适应质量调整
- 跨平台性能分析和比较
- 自动化性能测试和回归

记住：优化的目标不是追求极致的性能数字，而是在保持游戏体验的前提下，提供稳定流畅的运行表现。有时，为了艺术效果或游戏性，需要接受一定的性能开销。
